4. 迭代器（Iterator）与生成器（Generator）面试高频

// 迭代器协议：对象有 next() 方法，返回 { value, done } function createRange(start, end) { let current = start return { [Symbol.iterator]() { return this }, next() { return current <= end ? { value: current++, done: false } : { value: undefined, done: true } } } } console.log('自定义迭代器:', JSON.stringify([...createRange(1, 5)])) // 生成器函数（function*） function* fibonacci() { let [a, b] = [0, 1] while (true) { yield a; [a, b] = [b, a + b] } } const fib = fibonacci() const first8 = Array.from({ length: 8 }, () => fib.next().value) console.log('Fibonacci 前8项:', JSON.stringify(first8)) // 生成器实现 range function* range(start, end, step = 1) { for (let i = start; i <= end; i += step) yield i } console.log('range(1,10,2):', JSON.stringify([...range(1, 10, 2)])) // yield* 委托 function* concat(...iters) { for (const iter of iters) yield* iter } console.log('yield* concat:', JSON.stringify([...concat([1,2], [3,4], [5])])) // 生成器双向通信（send value via next） function* calculator() { let result = 0 while (true) { const input = yield result if (input === null) break result += input } return result } const calc = calculator() calc.next() // 启动 calc.next(10) // result = 10 calc.next(20) // result = 30 const final = calc.next(5) // result = 35 console.log('生成器双向通信 result:', final.value)

5. Proxy 与 Reflect 面试高频

Proxy 拦截对象的基本操作（读取、赋值、删除等），是 Vue3 响应式系统的核心。
// 数据验证代理 const validator = { set(target, key, value) { if (key === 'age' && (typeof value !== 'number' || value < 0)) throw new TypeError('age 必须是非负数字') target[key] = value return true } } const person = new Proxy({}, validator) person.age = 25 console.log('Proxy 验证通过 age:', person.age) try { person.age = -1 } catch(e) { console.warn('Proxy 验证失败:', e.message) } // 响应式代理（Vue3 原理简化版） const logs = [] function reactive(obj) { return new Proxy(obj, { get(target, key) { logs.push(`get:${key}`); return Reflect.get(target, key) }, set(target, key, value) { logs.push(`set:${key}=${value}`); return Reflect.set(target, key, value) } }) } const state = reactive({ count: 0, name: 'Alice' }) state.count++ const _name = state.name console.log('Proxy 响应式日志:', JSON.stringify(logs)) // 函数调用追踪 function traceFunction(fn, name) { return new Proxy(fn, { apply(target, thisArg, args) { console.log(`调用 ${name}(${args.join(',')})`) const result = Reflect.apply(target, thisArg, args) console.log(`${name} 返回:`, result) return result } }) } const tracedAdd = traceFunction((a, b) => a + b, 'add') tracedAdd(3, 4)
