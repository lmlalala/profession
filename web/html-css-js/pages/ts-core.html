1. 基础类型与类型注解 面试高频

TypeScript = JavaScript + 静态类型系统，编译时检查类型错误，运行时类型信息会被擦除。
// 基础类型注解 const name: string = 'Alice' const age: number = 25 const isDev: boolean = true const skills: string[] = ['TS', 'React', 'Vue'] const tuple: [string, number] = ['Alice', 25] // 元组：固定长度和类型 // 枚举 enum Direction { Up, Down, Left, Right } const dir: Direction = Direction.Up // 值为 0 // const 枚举（编译后内联，无运行时开销） const enum Status { Active = 'ACTIVE', Inactive = 'INACTIVE' } // any vs unknown vs never let a: any = 42 // 任意类型，跳过类型检查（慎用） let b: unknown = 42 // 安全的 any，使用前必须类型收窄 // let c: never // 永远不会有值（抛异常、死循环） // void vs undefined function log(msg: string): void { console.log(msg) } // 无返回值 let u: undefined = undefined // 类型断言 const el = document.getElementById('app') as HTMLDivElement const el2 = <HTMLDivElement>document.getElementById('app') // JSX 中不可用

2. interface 与 type 面试必考

// ── interface：描述对象形状，可声明合并 ── interface User { readonly id: number // 只读 name: string age?: number // 可选 greet(): string } // 声明合并（同名 interface 自动合并） interface User { email: string } // 现在 User 有 id, name, age?, greet(), email // 继承 interface Admin extends User { role: 'admin' | 'superadmin' } // ── type：类型别名，支持联合/交叉/元组/映射 ── type ID = string | number // 联合类型 type Point = { x: number; y: number } type Point3D = Point & { z: number } // 交叉类型 type Status = 'loading' | 'success' | 'error' // 字面量联合 // 函数类型 type Callback = (data: string) => void type AsyncFn<T> = () => Promise<T> // ── 核心区别 ── // interface: 可声明合并、extends 继承、描述对象/类 // type: 支持联合/交叉/元组/映射/条件类型，不可声明合并 // 推荐：对象用 interface，其他用 type
面试高频问题：interface 和 type 的区别？
① interface 可声明合并，type 不行
② type 支持联合类型、交叉类型、条件类型等高级特性
③ interface 可被 class implements
④ 性能上 interface 略优（编译器内部优化）

3. 泛型（Generics）面试必考

泛型 = 类型的参数化，让函数/类/接口在定义时不指定具体类型，使用时再确定。
// 泛型函数 function identity<T>(value: T): T { return value } identity<string>('hello') // 显式指定 identity(42) // 自动推断为 number // 泛型约束（extends） interface HasLength { length: number } function logLength<T extends HasLength>(value: T): number { return value.length } logLength('hello') // OK: string 有 length logLength([1, 2, 3]) // OK: array 有 length // logLength(42) // Error: number 没有 length // 多个泛型参数 function swap<A, B>(a: A, b: B): [B, A] { return [b, a] } // keyof + 泛型：安全地访问对象属性 function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] { return obj[key] } const user = { name: 'Alice', age: 25 } getProperty(user, 'name') // 返回类型自动推断为 string // getProperty(user, 'foo') // Error: 'foo' 不在 keyof User 中 // 泛型接口 interface ApiResponse<T> { code: number data: T message: string } type UserResponse = ApiResponse<User> // 泛型类 class Container<T> { private items: T[] = [] add(item: T) { this.items.push(item) } get(index: number): T { return this.items[index] } } // 泛型默认值 type Pagination<T = any> = { list: T[]; total: number; page: number }

4. 内置工具类型 面试高频

interface User { id: number name: string email: string age: number } // Partial<T> — 所有属性变可选 type UpdateUser = Partial<User> // { id?: number; name?: string; email?: string; age?: number } // Required<T> — 所有属性变必选 type StrictUser = Required<User> // Readonly<T> — 所有属性变只读 type FrozenUser = Readonly<User> // Pick<T, K> — 选取部分属性 type UserBasic = Pick<User, 'id' | 'name'> // { id: number; name: string } // Omit<T, K> — 排除部分属性 type UserWithoutEmail = Omit<User, 'email'> // { id: number; name: string; age: number } // Record<K, V> — 构造键值对类型 type RoleMap = Record<'admin' | 'user' | 'guest', string[]> // Extract<T, U> — 从联合类型中提取 type NumOrStr = Extract<string | number | boolean, string | number> // string | number // Exclude<T, U> — 从联合类型中排除 type OnlyBool = Exclude<string | number | boolean, string | number> // boolean // NonNullable<T> — 排除 null 和 undefined type SafeStr = NonNullable<string | null | undefined> // string // ReturnType<T> — 获取函数返回类型 function fetchUser() { return { id: 1, name: 'Alice' } } type FetchResult = ReturnType<typeof fetchUser> // { id: number; name: string } // Parameters<T> — 获取函数参数类型（元组） type FetchParams = Parameters<typeof fetchUser> // []
工具类型作用常见场景
Partial<T>全部可选更新接口、表单初始值
Required<T>全部必选严格校验
Readonly<T>全部只读不可变数据
Pick<T, K>选取属性API 响应裁剪
Omit<T, K>排除属性隐藏敏感字段
Record<K, V>键值映射字典、配置表
ReturnType<T>函数返回类型推断第三方库返回

5. 类型守卫与类型收窄 面试高频

// typeof 守卫 function padLeft(value: string, padding: string | number): string { if (typeof padding === 'number') { return ' '.repeat(padding) + value // 这里 padding 被收窄为 number } return padding + value // 这里 padding 被收窄为 string } // instanceof 守卫 class Dog { bark() { return 'woof' } } class Cat { meow() { return 'meow' } } function speak(animal: Dog | Cat) { if (animal instanceof Dog) { animal.bark() // OK: 收窄为 Dog } else { animal.meow() // OK: 收窄为 Cat } } // in 守卫 interface Fish { swim(): void } interface Bird { fly(): void } function move(animal: Fish | Bird) { if ('swim' in animal) { animal.swim() // 收窄为 Fish } else { animal.fly() // 收窄为 Bird } } // 自定义类型守卫（is 关键字） function isString(value: unknown): value is string { return typeof value === 'string' } function process(value: unknown) { if (isString(value)) { value.toUpperCase() // OK: 收窄为 string } } // 可辨识联合（Discriminated Union） type Shape = | { kind: 'circle'; radius: number } | { kind: 'rect'; width: number; height: number } function area(shape: Shape): number { switch (shape.kind) { case 'circle': return Math.PI * shape.radius ** 2 case 'rect': return shape.width * shape.height } } // never 穷尽检查 function assertNever(x: never): never { throw new Error('Unexpected: ' + x) } function areaStrict(shape: Shape): number { switch (shape.kind) { case 'circle': return Math.PI * shape.radius ** 2 case 'rect': return shape.width * shape.height default: return assertNever(shape) // 如果漏了 case，编译报错 } }

6. 高级类型与类型体操 面试高频

// ── 条件类型（Conditional Types）── type IsString<T> = T extends string ? 'yes' : 'no' type A = IsString<string> // 'yes' type B = IsString<number> // 'no' // ── infer 关键字：在条件类型中推断 ── type UnpackPromise<T> = T extends Promise<infer U> ? U : T type R1 = UnpackPromise<Promise<string>> // string type R2 = UnpackPromise<number> // number // 获取函数返回类型（手写 ReturnType） type MyReturnType<T> = T extends (...args: any[]) => infer R ? R : never // 获取数组元素类型 type ElementOf<T> = T extends (infer E)[] ? E : never type E1 = ElementOf<string[]> // string // ── 映射类型（Mapped Types）── type Optional<T> = { [K in keyof T]?: T[K] } type ReadonlyAll<T> = { readonly [K in keyof T]: T[K] } // 映射类型 + 条件类型 type StringKeys<T> = { [K in keyof T]: T[K] extends string ? K : never }[keyof T] // 提取值为 string 的 key // ── 模板字面量类型（Template Literal Types）── type EventName = `on${Capitalize<'click' | 'focus' | 'blur'>}` // 'onClick' | 'onFocus' | 'onBlur' type CSSProperty = `${string}-${string}` // 匹配 kebab-case // ── 递归类型 ── type DeepReadonly<T> = { readonly [K in keyof T]: T[K] extends object ? DeepReadonly<T[K]> : T[K] } // ── 实用类型体操 ── // Flatten：将嵌套数组类型拍平 type Flatten<T> = T extends (infer U)[] ? Flatten<U> : T type F1 = Flatten<number[][][]> // number
类型体操核心三件套：extends（条件判断）+ infer（类型推断）+ keyof/in（映射遍历）。掌握这三个就能应对绝大多数类型编程场景。
