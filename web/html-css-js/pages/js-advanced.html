2. 防抖（debounce）与节流（throttle）面试必考

防抖：最后一次触发后等待 n ms 才执行（搜索框输入）
节流：每 n ms 最多执行一次（滚动、resize）
/** * 防抖 —— 最后一次触发后等待 delay ms 才执行，支持立即执行模式。 * * @param {Function} fn - 要防抖的函数 * @param {number} delay - 延迟毫秒数 * @param {boolean} immediate - 是否首次立即执行 * @returns {Function} 防抖后的函数 */ function debounce(fn, delay, immediate = false) { let timer = null return function(...args) { if (timer) clearTimeout(timer) if (immediate && !timer) fn.apply(this, args) timer = setTimeout(() => { if (!immediate) fn.apply(this, args) timer = null }, delay) } } /** * 节流（时间戳版）—— 每 interval ms 最多执行一次。 * * @param {Function} fn - 要节流的函数 * @param {number} interval - 节流间隔毫秒数 * @returns {Function} 节流后的函数 */ function throttle(fn, interval) { let last = 0 return function(...args) { const now = Date.now() if (now - last >= interval) { last = now return fn.apply(this, args) } } } // 模拟测试 const logs = [] const debouncedFn = debounce(v => logs.push(`防抖执行：${v}`), 100) const throttledFn = throttle(v => logs.push(`节流执行：${v}`), 100) for (let i = 1; i <= 5; i++) { debouncedFn(`input${i}`) throttledFn(`click${i}`) } await new Promise(r => setTimeout(r, 150)) console.log('防抖结果（只执行最后一次）:', logs.filter(l => l.includes('防抖'))) console.log('节流结果（只执行第一次）:', logs.filter(l => l.includes('节流')))
/** * 防抖 —— 最后一次触发后等待 delay ms 才执行，支持立即执行模式。 * * @param fn - 要防抖的函数 * @param delay - 延迟毫秒数 * @param immediate - 是否首次立即执行 * @returns 防抖后的函数 */ function debounce<T extends (...args: unknown[]) => void>( fn: T, delay: number, immediate: boolean = false ): (...args: Parameters<T>) => void { let timer: ReturnType<typeof setTimeout> | null = null return function (this: unknown, ...args: Parameters<T>): void { if (timer) clearTimeout(timer) if (immediate && !timer) fn.apply(this, args) timer = setTimeout(() => { if (!immediate) fn.apply(this, args) timer = null }, delay) } } /** * 节流（时间戳版）—— 每 interval ms 最多执行一次。 * * @param fn - 要节流的函数 * @param interval - 节流间隔毫秒数 * @returns 节流后的函数 */ function throttle<T extends (...args: unknown[]) => unknown>( fn: T, interval: number ): (...args: Parameters<T>) => ReturnType<T> | undefined { let last = 0 return function (this: unknown, ...args: Parameters<T>) { const now = Date.now() if (now - last >= interval) { last = now return fn.apply(this, args) as ReturnType<T> } return undefined } }

/** * 柯里化 —— 将多参数函数转换为逐个接收参数的函数链。 * * @param {Function} fn - 要柯里化的函数 * @returns {Function} 柯里化后的函数 */ function curry(fn) { return function curried(...args) { if (args.length >= fn.length) return fn.apply(this, args) return (...more) => curried(...args, ...more) } } const add = curry((a, b, c) => a + b + c) console.log('curry(1)(2)(3):', add(1)(2)(3)) console.log('curry(1,2)(3):', add(1, 2)(3)) console.log('curry(1)(2,3):', add(1)(2, 3)) /** * 偏函数 —— 预先填充部分参数，返回接收剩余参数的新函数。 * * @param {Function} fn - 原始函数 * @param {...*} preArgs - 预填充的参数 * @returns {Function} 偏应用后的函数 */ function partial(fn, ...preArgs) { return (...laterArgs) => fn(...preArgs, ...laterArgs) } const multiply = (a, b, c) => a * b * c const double = partial(multiply, 2) console.log('偏函数 double(3,4):', double(3, 4)) /** * 函数组合（从右到左）。 * * @param {...Function} fns - 要组合的函数列表 * @returns {Function} 组合后的函数 */ const compose = (...fns) => x => fns.reduceRight((v, f) => f(v), x) /** * 管道（从左到右）。 * * @param {...Function} fns - 要串联的函数列表 * @returns {Function} 管道函数 */ const pipe = (...fns) => x => fns.reduce((v, f) => f(v), x) const double2 = x => x * 2 const addOne = x => x + 1 const square = x => x * x console.log('pipe(double,addOne,square)(3):', pipe(double2, addOne, square)(3)) console.log('compose(square,addOne,double)(3):', compose(square, addOne, double2)(3)) /** * 记忆化 —— 缓存函数计算结果，相同参数直接返回缓存值。 * * @param {Function} fn - 要记忆化的函数 * @returns {Function} 带缓存的函数 */ function memoize(fn) { const cache = new Map() return function(...args) { const key = JSON.stringify(args) if (cache.has(key)) { console.log(`缓存命中 key=${key}`) return cache.get(key) } const result = fn.apply(this, args) cache.set(key, result) return result } } const expensiveFib = memoize(function fib(n) { return n <= 1 ? n : expensiveFib(n - 1) + expensiveFib(n - 2) }) console.log('memoize fib(10):', expensiveFib(10)) console.log('memoize fib(10) 再次（命中缓存）:', expensiveFib(10))

6. 错误处理最佳实践

/** * 应用级错误基类，携带错误代码和附加数据。 */ class AppError extends Error { /** * @param {string} message - 错误消息 * @param {string} code - 错误代码 * @param {*} data - 附加数据 */ constructor(message, code, data) { super(message) this.name = 'AppError' this.code = code this.data = data } } /** 网络错误 */ class NetworkError extends AppError {} /** 验证错误 */ class ValidationError extends AppError {} try { throw new NetworkError('连接超时', 'TIMEOUT', { url: '/api/users' }) } catch(e) { console.log('错误类型:', e.name) console.log('错误消息:', e.message) console.log('错误代码:', e.code) console.log('是 NetworkError:', e instanceof NetworkError) console.log('是 AppError:', e instanceof AppError) console.log('是 Error:', e instanceof Error) } /** * Result 模式 —— 将 Promise 的 resolve/reject 转换为 [error, data] 元组。 * * @param {Promise} promise - 要包装的 Promise * @returns {Promise<[Error|null, *|null]>} [错误, 数据] 元组 */ async function safeAsync(promise) { try { return [null, await promise] } catch(e) { return [e, null] } } const delay = (ms, val, fail = false) => new Promise((res, rej) => setTimeout(() => (fail ? rej(new Error(val)) : res(val)), ms) ) const [err1, data1] = await safeAsync(delay(10, 'success')) console.log('Result 成功:', err1, data1) const [err2, data2] = await safeAsync(delay(10, '请求失败', true)) console.log('Result 失败:', err2?.message, data2)
/** * 应用级错误基类，携带错误代码和附加数据。 */ class AppError<D = unknown> extends Error { readonly code: string readonly data: D /** * @param message - 错误消息 * @param code - 错误代码 * @param data - 附加数据 */ constructor(message: string, code: string, data: D) { super(message) this.name = 'AppError' this.code = code this.data = data } } /** 网络错误 */ class NetworkError<D = unknown> extends AppError<D> { constructor(message: string, code: string, data: D) { super(message, code, data) this.name = 'NetworkError' } } /** 验证错误 */ class ValidationError<D = unknown> extends AppError<D> { constructor(message: string, code: string, data: D) { super(message, code, data) this.name = 'ValidationError' } } /** Result 元组类型：成功为 [null, T]，失败为 [Error, null] */ type Result<T> = [null, T] | [Error, null] /** * Result 模式 —— 将 Promise 的 resolve/reject 转换为 [error, data] 元组。 * * @param promise - 要包装的 Promise * @returns [错误, 数据] 元组 */ async function safeAsync<T>(promise: Promise<T>): Promise<Result<T>> { try { return [null, await promise] } catch (e) { return [e instanceof Error ? e : new Error(String(e)), null] } }
