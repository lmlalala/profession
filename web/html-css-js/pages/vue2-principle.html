1. 响应式原理：Object.defineProperty 面试必考

核心思路 ：遍历 data 的每个属性，用 Object.defineProperty 将其转为 getter/setter。 getter 中收集依赖（Dep.depend），setter 中通知更新（Dep.notify）。
/** * 依赖收集器，每个响应式属性对应一个 Dep 实例。 * 通过 depend 收集当前 Watcher，通过 notify 触发所有 Watcher 更新。 */ class Dep { constructor() { this.subs = new Set() } /** 将当前活跃的 Watcher 添加到订阅集合 */ depend() { if (Dep.target) this.subs.add(Dep.target) } /** 通知所有订阅的 Watcher 执行更新 */ notify() { this.subs.forEach(watcher => watcher.update()) } } Dep.target = null /** * 观察者，封装依赖收集与更新回调逻辑。 */ class Watcher { /** * @param {Object} vm - 关联的视图模型实例 * @param {Function|string} expOrFn - 取值函数或属性路径 * @param {Function} cb - 值变化时的回调函数 */ constructor(vm, expOrFn, cb) { this.vm = vm; this.cb = cb this.getter = typeof expOrFn === 'function' ? expOrFn : () => vm[expOrFn] this.value = this.get() } /** * 执行 getter 并收集依赖。 * @returns {*} getter 的返回值 */ get() { Dep.target = this const value = this.getter() Dep.target = null return value } /** 重新求值并触发回调 */ update() { const newVal = this.get() this.cb(newVal, this.value) this.value = newVal } } /** * 将对象的单个属性转为响应式。 * @param {Object} obj - 目标对象 * @param {string} key - 属性名 * @param {*} val - 属性初始值 */ function defineReactive(obj, key, val) { const dep = new Dep() if (val && typeof val === 'object') observe(val) Object.defineProperty(obj, key, { get() { dep.depend(); return val }, set(newVal) { if (newVal === val) return val = newVal if (typeof newVal === 'object') observe(newVal) dep.notify() } }) } /** * 递归遍历对象，将所有属性转为响应式。 * @param {Object} obj - 需要观察的对象 */ function observe(obj) { if (!obj || typeof obj !== 'object') return Object.keys(obj).forEach(key => defineReactive(obj, key, obj[key])) } // 测试响应式系统 const data = { name: 'Alice', age: 25, address: { city: 'Beijing' } } observe(data) new Watcher(data, () => data.name, (newVal, oldVal) => { console.log(`[Watcher] name: "${oldVal}" → "${newVal}"`) }) new Watcher(data, () => data.address.city, (newVal, oldVal) => { console.log(`[Watcher] address.city: "${oldVal}" → "${newVal}"`) }) console.log('初始 name:', data.name) data.name = 'Bob' data.name = 'Charlie' data.address.city = 'Shanghai' console.log('最终 name:', data.name)
/** Watcher 回调函数类型 */ type WatcherCallback<T> = (newVal: T, oldVal: T) => void /** 当前活跃的依赖收集目标 */ type DepTarget = Watcher<any> | null /** * 依赖收集器，每个响应式属性对应一个 Dep 实例。 */ class Dep { static target: DepTarget = null private subs: Set<Watcher<any>> = new Set() /** 将当前活跃的 Watcher 添加到订阅集合 */ depend(): void { if (Dep.target) this.subs.add(Dep.target) } /** 通知所有订阅的 Watcher 执行更新 */ notify(): void { this.subs.forEach(w => w.update()) } } /** * 观察者，封装依赖收集与更新回调逻辑。 */ class Watcher<T> { private vm: Record<string, any> private cb: WatcherCallback<T> private getter: () => T public value: T /** * @param vm - 关联的视图模型实例 * @param expOrFn - 取值函数或属性路径 * @param cb - 值变化时的回调函数 */ constructor(vm: Record<string, any>, expOrFn: (() => T) | string, cb: WatcherCallback<T>) { this.vm = vm; this.cb = cb this.getter = typeof expOrFn === 'function' ? expOrFn : () => vm[expOrFn] this.value = this.get() } /** * 执行 getter 并收集依赖。 * @returns getter 的返回值 */ get(): T { Dep.target = this const value = this.getter() Dep.target = null return value } /** 重新求值并触发回调 */ update(): void { const newVal = this.get() this.cb(newVal, this.value) this.value = newVal } } /** * 将对象的单个属性转为响应式。 * @param obj - 目标对象 * @param key - 属性名 * @param val - 属性初始值 */ function defineReactive(obj: Record<string, any>, key: string, val: any): void { const dep = new Dep() if (val && typeof val === 'object') observe(val) Object.defineProperty(obj, key, { get(): any { dep.depend(); return val }, set(newVal: any) { if (newVal === val) return val = newVal if (typeof newVal === 'object') observe(newVal) dep.notify() } }) } /** * 递归遍历对象，将所有属性转为响应式。 * @param obj - 需要观察的对象 */ function observe(obj: Record<string, any>): void { if (!obj || typeof obj !== 'object') return Object.keys(obj).forEach(key => defineReactive(obj, key, obj[key])) }

2. Vue2 响应式局限性 面试必考

Object.defineProperty 的缺陷 ：只能劫持已存在的属性，无法检测属性的新增和删除，也无法检测数组下标直接赋值。
// 依赖 Section 1 的 Dep / Watcher / defineReactive / observe（已通过 import 引入） const limData = { name: 'Alice', list: [1, 2, 3] } observe(limData) const limChanges = [] new Watcher(limData, () => limData.name, (n, o) => limChanges.push(`name: ${o}→${n}`)) // ✅ 正常响应：修改已有属性 limData.name = 'Bob' console.log('✅ 修改已有属性 name=Bob → 响应:', limChanges) // ❌ 无法检测：新增属性 const before = limChanges.length limData.newProp = 'new' console.log('❌ 新增属性 newProp → 无响应（changes 未增加）:', limChanges.length === before) // ❌ 无法检测：删除属性 const before2 = limChanges.length delete limData.name console.log('❌ 删除属性 name → 无响应:', limChanges.length === before2) // ❌ 数组下标赋值无法检测 limData.list[0] = 99 console.log('❌ limData.list[0] = 99 → 无响应（Vue2 缺陷）list:', JSON.stringify(limData.list)) console.log('\n💡 Vue2 解决方案：Vue.set(obj, key, val) / vm.$set') console.log('💡 Vue3 用 Proxy 彻底解决了这些问题')

3. 虚拟 DOM（Virtual DOM） 面试必考

核心思想 ：用 JS 对象描述 DOM 结构（VNode），数据变化时先对比新旧 VNode（diff），再将最小差异更新到真实 DOM。
/** * 创建虚拟 DOM 节点（VNode）。 * @param {string} tag - 标签名 * @param {Object|null} props - 属性对象 * @param {...(Object|string)} children - 子节点列表 * @returns {Object} VNode 对象 */ function h(tag, props, ...children) { return { tag, props: props || {}, children: children.flat().filter(c => c != null) } } /** * 将 VNode 递归转为真实 DOM 节点。 * @param {Object|string} vnode - VNode 或文本节点 * @returns {Node} 真实 DOM 节点 */ function createEl(vnode) { if (typeof vnode === 'string' || typeof vnode === 'number') return document.createTextNode(String(vnode)) const el = document.createElement(vnode.tag) Object.entries(vnode.props).forEach(([k, v]) => { if (k === 'class') el.className = v else if (k === 'style' && typeof v === 'object') Object.assign(el.style, v) else if (typeof v === 'function') el.addEventListener(k.replace(/^on/, '').toLowerCase(), v) else if (k !== 'key') el.setAttribute(k, v) }) vnode.children.forEach(child => el.appendChild(createEl(child))) return el } // ── 构建 VNode 树 ───────────────────────────────────────────── const vnode = h('div', { id: 'app' }, h('h1', null, 'Hello Vue'), h('ul', null, h('li', { key: 'a' }, 'Item A'), h('li', { key: 'b' }, 'Item B'), h('li', { key: 'c' }, 'Item C'), ) ) // ── ① 打印 VNode 对象结构（JS 对象，非真实 DOM）──────────────── const pre = document.createElement('pre') pre.style.cssText = 'background:#161b2e;color:#8b85ff;padding:0.75rem;border-radius:4px;font-size:0.8rem;margin-bottom:1rem;overflow:auto' function vnodeToText(node, depth = 0) { const indent = ' '.repeat(depth) if (typeof node === 'string') return `${indent}"${node}"\n` const propsStr = Object.keys(node.props).length ? ' props=' + JSON.stringify(node.props) : '' let s = `${indent}{ tag: "${node.tag}"${propsStr} }\n` node.children.forEach(c => { s += vnodeToText(c, depth + 1) }) return s } pre.textContent = '── VNode 对象结构（h 函数返回值）──\n\n' + vnodeToText(vnode) document.body.appendChild(pre) // ── ② 将 VNode 挂载为真实 DOM ──────────────────────────────── const label = document.createElement('div') label.style.cssText = 'font-size:0.75rem;color:#8892b0;margin-bottom:0.5rem' label.textContent = '── 挂载为真实 DOM（createEl 输出）──' document.body.appendChild(label) const domEl = createEl(vnode) document.body.appendChild(domEl)
/** VNode 属性类型 */ interface VNodeProps { [key: string]: string | number | boolean | Record<string, string> | ((...a: any[]) => void) } /** 虚拟 DOM 节点 */ interface VNode { tag: string props: VNodeProps children: Array<VNode | string> } /** * 创建虚拟 DOM 节点（VNode）。 */ function h(tag: string, props: VNodeProps | null, ...children: Array<VNode | string | null>): VNode { return { tag, props: props || {}, children: children.flat().filter((c): c is VNode | string => c != null) } } /** * 将 VNode 递归转为真实 DOM 节点。 */ function createEl(vnode: VNode | string): Node { if (typeof vnode === 'string') return document.createTextNode(vnode) const el = document.createElement(vnode.tag) Object.entries(vnode.props).forEach(([k, v]) => { if (k === 'class') el.className = v as string else if (k === 'style' && typeof v === 'object') Object.assign(el.style, v) else if (typeof v === 'function') el.addEventListener(k.replace(/^on/, '').toLowerCase(), v as EventListener) else if (k !== 'key') el.setAttribute(k, String(v)) }) vnode.children.forEach(child => el.appendChild(createEl(child))) return el } // 使用示例（类型推断） const vnode: VNode = h('div', { id: 'app' }, h('h1', null, 'Hello Vue'), h('ul', null, h('li', { key: 'a' }, 'Item A'), h('li', { key: 'b' }, 'Item B'), h('li', { key: 'c' }, 'Item C'), ) ) // createEl 返回 Node，可直接 appendChild const el: Node = createEl(vnode)

4. diff 算法（同层对比） 面试必考

Vue2 diff 策略 ：同层比较，不跨层。使用双端指针（头头/尾尾/头尾/尾头）对比，再用 key 建立映射表。 时间复杂度 O(n)。
/** * Vue2 双端指针 diff 算法（简化版，操作 key 字符串列表）。 * 策略：头头/尾尾/头尾/尾头 四种对比，命中则复用并移动指针； * 都不命中时用 key→index 映射表 O(1) 查找旧节点。 * * @param {string[]} oldList - 旧子节点 key 列表 * @param {string[]} newList - 新子节点 key 列表 * @returns {{ ops: string[], result: string[] }} 操作日志与最终顺序 */ function vue2Diff(oldList, newList) { // 用数组模拟可变的旧列表（patch 时会移动/删除元素） const old = [...oldList] const ops = [] let oldStart = 0, oldEnd = old.length - 1 let newStart = 0, newEnd = newList.length - 1 while (oldStart <= oldEnd && newStart <= newEnd) { const oS = old[oldStart], oE = old[oldEnd] const nS = newList[newStart], nE = newList[newEnd] if (oS === undefined) { oldStart++; continue } if (oE === undefined) { oldEnd--; continue } if (oS === nS) { // ① 头头相同：直接复用，双端指针同时向内移动 ops.push(`复用 ${nS}（头头）`) oldStart++; newStart++ } else if (oE === nE) { // ② 尾尾相同 ops.push(`复用 ${nE}（尾尾）`) oldEnd--; newEnd-- } else if (oS === nE) { // ③ 旧头 = 新尾：旧头节点移到旧尾之后 ops.push(`移动 ${oS}（旧头→新尾）`) oldStart++; newEnd-- } else if (oE === nS) { // ④ 旧尾 = 新头：旧尾节点移到旧头之前 ops.push(`移动 ${oE}（旧尾→新头）`) oldEnd--; newStart++ } else { // ⑤ 四种都不命中：用 key 映射表查找 const keyMap = new Map(old.map((k, i) => [k, i])) const idxInOld = keyMap.get(nS) if (idxInOld !== undefined) { ops.push(`移动 ${nS}（key 映射）`) old[idxInOld] = undefined // 标记已处理 } else { ops.push(`新增 ${nS}`) } newStart++ } } // 处理剩余节点 while (newStart <= newEnd) { ops.push(`新增 ${newList[newStart++]}`) } while (oldStart <= oldEnd) { if (old[oldStart] !== undefined) ops.push(`删除 ${old[oldStart]}`) oldStart++ } return { ops, result: newList } } // 测试 1：纯移动（无增删） const r1 = vue2Diff(['A','B','C','D','E'], ['A','C','E','B','D']) console.log('=== 测试1：纯移动 ===') console.log('旧: A B C D E → 新: A C E B D') r1.ops.forEach(op => console.log(' ', op)) // 测试 2：有新增和删除 const r2 = vue2Diff(['A','B','C'], ['B','C','D','E']) console.log('\n=== 测试2：增删 ===') console.log('旧: A B C → 新: B C D E') r2.ops.forEach(op => console.log(' ', op)) // 测试 3：头尾互换（经典双端命中场景） const r3 = vue2Diff(['A','B','C','D'], ['D','C','B','A']) console.log('\n=== 测试3：完全反转 ===') console.log('旧: A B C D → 新: D C B A') r3.ops.forEach(op => console.log(' ', op))
/** 双端 diff 操作日志与结果 */ interface DiffResult { /** 每一步操作的文字描述 */ ops: string[] /** diff 后的最终顺序（即 newList） */ result: string[] } /** * Vue2 双端指针 diff 算法（简化版，操作 key 字符串列表）。 * 策略：头头/尾尾/头尾/尾头 四种对比，命中则复用并移动指针； * 都不命中时用 key→index 映射表 O(1) 查找旧节点。 * * @param oldList - 旧子节点 key 列表 * @param newList - 新子节点 key 列表 * @returns 操作日志与最终顺序 */ function vue2Diff(oldList: string[], newList: string[]): DiffResult { const old: (string | undefined)[] = [...oldList] const ops: string[] = [] let oldStart = 0, oldEnd = old.length - 1 let newStart = 0, newEnd = newList.length - 1 while (oldStart <= oldEnd && newStart <= newEnd) { const oS = old[oldStart], oE = old[oldEnd] const nS = newList[newStart], nE = newList[newEnd] if (oS === undefined) { oldStart++; continue } if (oE === undefined) { oldEnd--; continue } if (oS === nS) { ops.push(`复用 ${nS}（头头）`) oldStart++; newStart++ } else if (oE === nE) { ops.push(`复用 ${nE}（尾尾）`) oldEnd--; newEnd-- } else if (oS === nE) { ops.push(`移动 ${oS}（旧头→新尾）`) oldStart++; newEnd-- } else if (oE === nS) { ops.push(`移动 ${oE}（旧尾→新头）`) oldEnd--; newStart++ } else { const keyMap = new Map<string, number>( old.map((k, i) => [k as string, i] as [string, number]) .filter(([k]) => k !== undefined) ) const idxInOld = keyMap.get(nS) if (idxInOld !== undefined) { ops.push(`移动 ${nS}（key 映射）`) old[idxInOld] = undefined } else { ops.push(`新增 ${nS}`) } newStart++ } } while (newStart <= newEnd) { ops.push(`新增 ${newList[newStart++]}`) } while (oldStart <= oldEnd) { if (old[oldStart] !== undefined) ops.push(`删除 ${old[oldStart]}`) oldStart++ } return { ops, result: newList } } // 测试 1：纯移动 const r1 = vue2Diff(['A','B','C','D','E'], ['A','C','E','B','D']) console.log('=== 测试1：纯移动 ===') r1.ops.forEach(op => console.log(' ', op)) // 测试 2：有新增和删除 const r2 = vue2Diff(['A','B','C'], ['B','C','D','E']) console.log('\n=== 测试2：增删 ===') r2.ops.forEach(op => console.log(' ', op))

5. Mini Vue2 完整实现

/** 依赖收集器 */ class Dep { constructor() { this.subs = new Set() } /** 收集当前 Watcher */ depend() { if (Dep.target) this.subs.add(Dep.target) } /** 通知所有 Watcher 更新 */ notify() { this.subs.forEach(w => w.update()) } } Dep.target = null /** * 观察者，支持 lazy 模式（用于 computed）。 */ class Watcher { /** * @param {Object} vm - Vue 实例 * @param {Function} getter - 取值函数 * @param {Function} cb - 变化回调 * @param {boolean} [lazy=false] - 是否延迟求值（computed 使用） */ constructor(vm, getter, cb, lazy = false) { this.vm = vm; this.cb = cb; this.getter = getter; this.lazy = lazy; this.dirty = lazy this.value = lazy ? undefined : this.get() } /** @returns {*} getter 返回值 */ get() { Dep.target = this; const v = this.getter.call(this.vm); Dep.target = null; return v } /** 触发更新，lazy 模式下仅标记 dirty */ update() { if (this.lazy) { this.dirty = true } else { const n = this.get(); this.cb(n, this.value); this.value = n } } /** 手动求值并清除 dirty 标记（computed 专用） */ evaluate() { this.value = this.get(); this.dirty = false } } /** * 将对象单个属性转为响应式。 * @param {Object} obj - 目标对象 * @param {string} key - 属性名 * @param {*} val - 初始值 */ function defineReactive(obj, key, val) { const dep = new Dep() if (val && typeof val === 'object') observe(val) Object.defineProperty(obj, key, { get() { dep.depend(); return val }, set(newVal) { if (newVal !== val) { val = newVal; dep.notify() } } }) } /** * 递归将对象所有属性转为响应式。 * @param {Object} obj - 目标对象 */ function observe(obj) { if (!obj || typeof obj !== 'object') return Object.keys(obj).forEach(key => defineReactive(obj, key, obj[key])) } /** * Mini Vue2 核心类，实现响应式 data、computed、watch、methods。 */ class MiniVue { /** * @param {Object} options - 配置项 * @param {Function|Object} options.data - 数据（函数或对象） * @param {Object} [options.computed] - 计算属性 * @param {Object} [options.watch] - 侦听器 * @param {Object} [options.methods] - 方法 * @param {Function} [options.created] - 生命周期钩子 */ constructor(options) { this.$data = typeof options.data === 'function' ?
