1. Proxy 响应式 vs Vue2 Object.defineProperty 面试必考

Vue3 用 Proxy 彻底解决了 Vue2 的缺陷：可以拦截属性新增/删除、数组下标赋值、Map/Set 操作，无需递归初始化（懒代理）。
特性Vue2 (defineProperty)Vue3 (Proxy)
属性新增❌ 需 Vue.set()✅ 自动检测
属性删除❌ 需 Vue.delete()✅ 自动检测
数组下标赋值❌ 不响应✅ 自动检测
初始化性能❌ 递归遍历所有属性✅ 懒代理（访问时才代理）
Map/Set 支持❌ 不支持✅ 原生支持
兼容性✅ IE9+❌ 不支持 IE
// Vue3 响应式核心：Proxy + Reflect + track/trigger const targetMap = new WeakMap() let activeEffect = null const effectStack = [] function track(target, key) { if (!activeEffect) return let depsMap = targetMap.get(target) if (!depsMap) targetMap.set(target, depsMap = new Map()) let dep = depsMap.get(key) if (!dep) depsMap.set(key, dep = new Set()) dep.add(activeEffect) } function trigger(target, type, key) { const depsMap = targetMap.get(target) if (!depsMap) return const effects = new Set() depsMap.get(key)?.forEach(e => { if (e !== activeEffect) effects.add(e) }) if (type === 'ADD' || type === 'DELETE') depsMap.get('length')?.forEach(e => effects.add(e)) effects.forEach(e => e()) } function effect(fn) { const effectFn = () => { effectStack.push(effectFn); activeEffect = effectFn const result = fn() effectStack.pop(); activeEffect = effectStack[effectStack.length - 1] ?? null return result } effectFn.deps = new Set() effectFn() return effectFn } const reactiveMap = new WeakMap() function reactive(raw) { if (reactiveMap.has(raw)) return reactiveMap.get(raw) const proxy = new Proxy(raw, { get(target, key, receiver) { track(target, key) const res = Reflect.get(target, key, receiver) return res && typeof res === 'object' ? reactive(res) : res }, set(target, key, value, receiver) { const hadKey = Object.hasOwn(target, key) const oldVal = target[key] const result = Reflect.set(target, key, value, receiver) if (!hadKey) trigger(target, 'ADD', key) else if (value !== oldVal) trigger(target, 'SET', key) return result }, deleteProperty(target, key) { const hadKey = Object.hasOwn(target, key) const result = Reflect.deleteProperty(target, key) if (hadKey) trigger(target, 'DELETE', key) return result } }) reactiveMap.set(raw, proxy) return proxy } // ── 演示 ── const state = reactive({ name: 'Alice', age: 25 }) const logs = [] effect(() => { logs.push(`name=${state.name}`) }) state.name = 'Bob' console.log('修改已有属性 name=Bob → 响应:', logs.join(' | ')) // ✅ 新增属性（Vue2 无法检测，Vue3 可以） const logs2 = [] effect(() => { logs2.push(`city=${state.city ?? 'undefined'}`) }) state.city = 'Beijing' console.log('✅ 新增属性 city=Beijing → 响应:', logs2.join(' | ')) // ✅ 删除属性 const logs3 = [] effect(() => { logs3.push(`age=${state.age ?? 'deleted'}`) }) delete state.age console.log('✅ 删除属性 age → 响应:', logs3.join(' | ')) // ✅ 数组下标赋值 const arr = reactive([1, 2, 3]) const arrLogs = [] effect(() => { arrLogs.push(`arr[0]=${arr[0]}`) }) arr[0] = 99 console.log('✅ 数组下标赋值 arr[0]=99 → 响应:', arrLogs.join(' | ')) // ✅ 深层嵌套（懒代理） const nested = reactive({ a: { b: { c: 1 } } }) const nestedLogs = [] effect(() => { nestedLogs.push(`c=${nested.a.b.c}`) }) nested.a.b.c = 42 console.log('✅ 深层嵌套 nested.a.b.c=42 → 响应:', nestedLogs.join(' | '))

2. effect / track / trigger 依赖收集 面试必考

// 全局依赖图：WeakMap<target, Map<key, Set<effect>>> const targetMap = new WeakMap() let activeEffect = null const effectStack = [] // track：在 getter 中收集当前 activeEffect function track(target, key) { if (!activeEffect) return let depsMap = targetMap.get(target) if (!depsMap) targetMap.set(target, depsMap = new Map()) let dep = depsMap.get(key) if (!dep) depsMap.set(key, dep = new Set()) dep.add(activeEffect) } // trigger：在 setter 中触发所有依赖该属性的 effect function trigger(target, type, key) { const depsMap = targetMap.get(target) if (!depsMap) return const effects = new Set() depsMap.get(key)?.forEach(e => { if (e !== activeEffect) effects.add(e) }) effects.forEach(e => e.options?.scheduler ? e.options.scheduler(e) : e()) } // effect：注册副作用函数，执行时自动收集依赖 function effect(fn, options = {}) { const effectFn = () => { effectStack.push(effectFn); activeEffect = effectFn const result = fn() effectStack.pop(); activeEffect = effectStack[effectStack.length - 1] ?? null return result } effectFn.deps = new Set() effectFn.options = options if (!options.lazy) effectFn() return effectFn } // reactive（简化版） function reactive(raw) { return new Proxy(raw, { get(target, key, receiver) { track(target, key) return Reflect.get(target, key, receiver) }, set(target, key, value, receiver) { const result = Reflect.set(target, key, value, receiver) trigger(target, 'SET', key) return result } }) } // ── 演示：effect track/trigger ── const state = reactive({ a: 1, b: 2, c: 3 }) const e1Logs = [] effect(() => { e1Logs.push(state.a + state.b) }) console.log('effect1 依赖 a+b，初始值:', e1Logs.at(-1)) const e2Logs = [] effect(() => { e2Logs.push(state.b + state.c) }) console.log('effect2 依赖 b+c，初始值:', e2Logs.at(-1)) // 修改 a → 只触发 effect1 state.a = 10 console.log('修改 a=10 → effect1 触发:', e1Logs.at(-1), '| effect2 未触发:', e2Logs.length === 1) // 修改 b → 触发 effect1 和 effect2 state.b = 20 console.log('修改 b=20 → effect1:', e1Logs.at(-1), '| effect2:', e2Logs.at(-1)) // 修改 c → 只触发 effect2 state.c = 30 console.log('修改 c=30 → effect2 触发:', e2Logs.at(-1), '| effect1 未触发:', e1Logs.length === 3) console.log('\n依赖图：a→[effect1] | b→[effect1, effect2] | c→[effect2]')

3. ref / reactive / computed 实现

// 依赖收集基础（复用 track/trigger/effect） const targetMap = new WeakMap() let activeEffect = null const effectStack = [] function track(target, key) { if (!activeEffect) return let depsMap = targetMap.get(target) if (!depsMap) targetMap.set(target, depsMap = new Map()) let dep = depsMap.get(key) if (!dep) depsMap.set(key, dep = new Set()) dep.add(activeEffect) } function trigger(target, type, key) { const depsMap = targetMap.get(target) if (!depsMap) return const effects = new Set() depsMap.get(key)?.forEach(e => { if (e !== activeEffect) effects.add(e) }) effects.forEach(e => e.options?.scheduler ? e.options.scheduler(e) : e()) } function effect(fn, options = {}) { const effectFn = () => { effectStack.push(effectFn); activeEffect = effectFn const result = fn() effectStack.pop(); activeEffect = effectStack[effectStack.length - 1] ?? null return result } effectFn.deps = new Set(); effectFn.options = options if (!options.lazy) effectFn() return effectFn } // ref：包装基本类型（通过 .value 访问） function ref(raw) { const r = { __v_isRef: true, get value() { track(r, 'value'); return raw }, set value(newVal) { if (newVal !== raw) { raw = newVal; trigger(r, 'SET', 'value') } } } return r } // computed：带缓存的计算属性 function computed(getter) { let dirty = true, _value const effectFn = effect(getter, { lazy: true, scheduler() { if (!dirty) { dirty = true; trigger(obj, 'SET', 'value') } } }) const obj = { __v_isRef: true, get value() { if (dirty) { _value = effectFn(); dirty = false } // 脏时重算 track(obj, 'value') return _value } } return obj } // watch：监听响应式数据变化 function watch(source, cb, options = {}) { let getter = typeof source === 'function' ? source : () => source.value let oldValue const job = () => { const newValue = effectFn() if (newValue !== oldValue) { cb(newValue, oldValue); oldValue = newValue } } const effectFn = effect(getter, { lazy: true, scheduler: job }) if (options.immediate) job() else oldValue = effectFn() } // ── 演示 ── // ref const count = ref(0) const logs = [] effect(() => { logs.push(`count=${count.value}`) }) count.value++; count.value++ console.log('ref count 变化:', logs.join(' → ')) // computed（带缓存） const price = ref(100) const quantity = ref(3) let computeCount = 0 const total = computed(() => { computeCount++; return price.value * quantity.value }) console.log('computed total:', total.value, '(计算次数:', computeCount, ')') console.log('再次读取 total（缓存）:', total.value, '(计算次数:', computeCount, ')') price.value = 200 console.log('price 变化后 total:', total.value, '(计算次数:', computeCount, ')') console.log('再次读取（缓存）:', total.value, '(计算次数:', computeCount, ')') // watch const name = ref('Alice') const watchLogs = [] watch(name, (newVal, oldVal) => { watchLogs.push(`${oldVal}→${newVal}`) }) name.value = 'Bob'; name.value = 'Charlie' console.log('watch 变化记录:', watchLogs.join(' | '))

4. Composition API vs Options API 面试高频

Vue2 Options API（按选项组织）
export default { data() { return { count: 0, name: '' } }, computed: { double() { return this.count * 2 } }, methods: { increment() { this.count++ } }, watch: { count(val) { console.log(val) } }, mounted() { /* 生命周期 */ } }
Vue3 Composition API（按逻辑组织）
import { ref, computed, watch, onMounted } from 'vue' export default { setup() { // 逻辑可抽取为 composable const count = ref(0) const double = computed(() => count.value * 2) const increment = () => count.value++ watch(count, val => console.log(val)) onMounted(() => { /* 生命周期 */ }) return { count, double, increment } } }
Composition API 优势： ① 相关逻辑聚合在一起（不像 Options API 分散在 data/methods/computed）； ② 逻辑可复用（抽取为 composable 函数，类似 React Hooks）； ③ 更好的 TypeScript 支持； ④ 树摇（Tree-shaking）友好

// Vue3 编译优化演示 // ① PatchFlag 模拟：标记动态内容类型，diff 时跳过静态部分 const PatchFlags = { TEXT: 1, CLASS: 2, STYLE: 4, PROPS: 8, FULL_PROPS: 16 } function createVNode(type, props, children, patchFlag = 0, dynamicProps = null) { return { type, props, children, patchFlag, dynamicProps, _isVNode: true } } const vnode1 = createVNode('div', { id: 'static' }, 'static text', 0) const vnode2 = createVNode('div', { class: 'dynamic' }, 'hello', PatchFlags.TEXT) const vnode3 = createVNode('div', { class: 'x' }, null, PatchFlags.CLASS) console.log('静态节点 patchFlag:', vnode1.patchFlag, '(0=无需diff)') console.log('动态文本 patchFlag:', vnode2.patchFlag, '(1=只diff文本)') console.log('动态class patchFlag:', vnode3.patchFlag, '(2=只diff class)') // ② 静态提升模拟：静态节点只创建一次 let renderCount = 0 const _hoisted = createVNode('span', null, '静态内容') // 只创建一次 function render(dynamic) { renderCount++ return createVNode('div', null, [_hoisted, createVNode('span', null, dynamic, PatchFlags.TEXT)]) } render('第一次'); render('第二次'); render('第三次') console.log(`静态提升：渲染 3 次，静态节点只创建 1 次（renderCount=${renderCount}）`) // ③ Block Tree 模拟：收集动态节点，diff 时只对比动态节点 const dynamicNodes = [] function openBlock() { dynamicNodes.length = 0 } function closeBlock(vnode) { vnode.dynamicChildren = [...dynamicNodes]; return vnode } function trackDynamic(vnode) { if (vnode.patchFlag) dynamicNodes.push(vnode); return vnode } openBlock() const block = closeBlock(createVNode('div', null, [ trackDynamic(createVNode('span', null, '静态', 0)), trackDynamic(createVNode('span', null, '动态1', PatchFlags.TEXT)), trackDynamic(createVNode('span', null, '动态2', PatchFlags.CLASS)), ])) console.log('Block Tree 动态子节点数:', block.dynamicChildren.length, '(跳过静态节点)') console.log('动态节点 patchFlags:', block.dynamicChildren.map(n => n.patchFlag).join(', '))
