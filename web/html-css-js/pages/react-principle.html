1. JSX 与虚拟 DOM 面试必考

JSX 本质：语法糖，Babel 将 JSX 编译为 React.createElement() 调用，返回描述 UI 的普通 JS 对象（VNode）。
JSX 写法
// 开发者写的 JSX const element = ( <div className="app"> <h1>Hello React</h1> <p style={{color:'red'}}> {message} </p> </div> )
Babel 编译结果
// Babel 编译后 const element = React.createElement( 'div', { className: 'app' }, React.createElement('h1', null, 'Hello React'), React.createElement('p', { style: { color: 'red' } }, message ) )
// createElement 返回 VNode 对象 function createElement(type, props, ...children) { return { type, // 'div' | 'p' | MyComponent props: { ...(props || {}), children: children.length === 1 ? children[0] : children }, key: props?.key ?? null, ref: props?.ref ?? null, $$typeof: Symbol.for('react.element') // 防 XSS } }

2. Fiber 架构 面试必考

为什么需要 Fiber？ React 15 的 Stack Reconciler 是同步递归的，一旦开始无法中断，大型组件树会阻塞主线程导致卡顿。 React 16 引入 Fiber，将渲染工作拆分为可中断的小单元，利用浏览器空闲时间（requestIdleCallback）执行。
// Fiber 节点结构（每个组件/DOM 对应一个 Fiber） const fiber = { type: 'div', // 组件类型 props: {}, // 属性 stateNode: domElement, // 对应的真实 DOM // 树结构（三指针） child: childFiber, // 第一个子节点 sibling: siblingFiber, // 下一个兄弟节点 return: parentFiber, // 父节点 // 双缓冲 alternate: workInProgress, // 对应的 work-in-progress fiber // 副作用 effectTag: 'PLACEMENT', // 'PLACEMENT' | 'UPDATE' | 'DELETION' // Hooks 链表 memoizedState: hookState, // 存储 Hooks 状态链表 } // 工作循环（可中断） function workLoop(deadline) { while (nextUnitOfWork && deadline.timeRemaining() > 1) { nextUnitOfWork = performUnitOfWork(nextUnitOfWork) } if (!nextUnitOfWork && wipRoot) commitRoot() // 全部完成后提交 requestIdleCallback(workLoop) // 继续调度 }

3. React diff 算法 面试必考

React diff 三大策略： ① 同层比较（不跨层）； ② 类型不同直接替换； ③ key 相同则复用（key 的重要性！）
// React 多节点 diff（React 18 核心逻辑） function reconcileChildrenArray(returnFiber, currentFirstChild, newChildren) { let resultFirstChild = null let previousNewFiber = null let oldFiber = currentFirstChild let lastPlacedIndex = 0 // 最后一个可复用节点在旧列表中的位置 let newIdx = 0 // 第一轮：顺序遍历，处理前缀相同的节点 for (; oldFiber && newIdx < newChildren.length; newIdx++) { const newChild = newChildren[newIdx] if (oldFiber.key !== newChild.key) break // key 不同，退出第一轮 const newFiber = updateElement(returnFiber, oldFiber, newChild) oldFiber = oldFiber.sibling } // 第二轮：剩余旧节点建立 key→fiber 映射表 const existingChildren = new Map() while (oldFiber) { existingChildren.set(oldFiber.key ?? oldFiber.index, oldFiber) oldFiber = oldFiber.sibling } // 遍历剩余新节点，查找可复用的旧节点 for (; newIdx < newChildren.length; newIdx++) { const newChild = newChildren[newIdx] const matchedFiber = existingChildren.get(newChild.key ?? newIdx) if (matchedFiber) { existingChildren.delete(newChild.key ?? newIdx) if (matchedFiber.index < lastPlacedIndex) // 需要移动 newFiber.effectTag = 'PLACEMENT' else lastPlacedIndex = matchedFiber.index // 不需要移动 } } // 剩余旧节点全部删除 existingChildren.forEach(child => deleteChild(returnFiber, child)) }

4. Hooks 原理实现 面试必考

Hooks 本质：每个组件对应一个 Fiber，Fiber 上有一个 memoizedState 链表，每个 Hook 调用对应链表中的一个节点。 这就是为什么 Hooks 不能在条件语句中调用——顺序必须固定！
// Hooks 链表结构 // fiber.memoizedState → hook1 → hook2 → hook3 → null const hook = { memoizedState: value, // 当前状态值 queue: { // 更新队列 pending: null, dispatch: null, }, next: nextHook // 指向下一个 Hook } // useState 实现 function useState(initialState) { const hook = getOrCreateHook() if (hook.memoizedState === undefined) hook.memoizedState = typeof initialState === 'function' ? initialState() : initialState const dispatch = (action) => { hook.memoizedState = typeof action === 'function' ? action(hook.memoizedState) : action scheduleUpdate() // 触发重新渲染 } return [hook.memoizedState, dispatch] } // useEffect 实现 function useEffect(callback, deps) { const hook = getOrCreateHook() const prevDeps = hook.memoizedState?.deps const hasChanged = !prevDeps || deps.some((d, i) => d !== prevDeps[i]) if (hasChanged) { hook.memoizedState?.cleanup?.() // 执行上次的清理函数 scheduleEffect(() => { const cleanup = callback() hook.memoizedState = { deps, cleanup } }) } }
