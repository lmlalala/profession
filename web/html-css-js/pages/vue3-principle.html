1. Proxy 响应式 vs Vue2 Object.defineProperty 面试必考

Vue3 用 Proxy 彻底解决了 Vue2 的缺陷：可以拦截属性新增/删除、数组下标赋值、Map/Set 操作，无需递归初始化（懒代理）。
特性Vue2 (defineProperty)Vue3 (Proxy)
属性新增❌ 需 Vue.set()✅ 自动检测
属性删除❌ 需 Vue.delete()✅ 自动检测
数组下标赋值❌ 不响应✅ 自动检测
初始化性能❌ 递归遍历所有属性✅ 懒代理（访问时才代理）
Map/Set 支持❌ 不支持✅ 原生支持
兼容性✅ IE9+❌ 不支持 IE
// Vue3 响应式核心：Proxy + Reflect function reactive(raw) { return new Proxy(raw, { get(target, key, receiver) { track(target, key) // 收集依赖 const res = Reflect.get(target, key, receiver) // 懒代理：访问嵌套对象时才代理 return res && typeof res === 'object' ? reactive(res) : res }, set(target, key, value, receiver) { const hadKey = Object.hasOwn(target, key) const result = Reflect.set(target, key, value, receiver) if (!hadKey) trigger(target, 'ADD', key) // 新增属性 else trigger(target, 'SET', key) // 修改属性 return result }, deleteProperty(target, key) { const result = Reflect.deleteProperty(target, key) trigger(target, 'DELETE', key) // 删除属性 return result } }) }

2. effect / track / trigger 依赖收集 面试必考

// 全局依赖图：WeakMap<target, Map<key, Set<effect>>> const targetMap = new WeakMap() let activeEffect = null // track：在 getter 中收集当前 activeEffect function track(target, key) { if (!activeEffect) return let depsMap = targetMap.get(target) if (!depsMap) targetMap.set(target, depsMap = new Map()) let dep = depsMap.get(key) if (!dep) depsMap.set(key, dep = new Set()) dep.add(activeEffect) } // trigger：在 setter 中触发所有依赖该属性的 effect function trigger(target, key) { const depsMap = targetMap.get(target) if (!depsMap) return depsMap.get(key)?.forEach(effect => effect()) } // effect：注册副作用函数，执行时自动收集依赖 function effect(fn) { const effectFn = () => { activeEffect = effectFn const result = fn() activeEffect = null return result } effectFn() // 立即执行，触发 getter，完成依赖收集 return effectFn }

3. ref / reactive / computed 实现

// ref：包装基本类型（通过 .value 访问） function ref(raw) { return { get value() { track(this, 'value'); return raw }, set value(newVal) { raw = newVal; trigger(this, 'value') }, __v_isRef: true } } // computed：带缓存的计算属性 function computed(getter) { let dirty = true, _value const effectFn = effect(getter, { lazy: true, scheduler() { dirty = true; trigger(obj, 'value') } // 依赖变化时标脏 }) const obj = { get value() { if (dirty) { _value = effectFn(); dirty = false } // 脏时重算 track(obj, 'value') return _value } } return obj } // watch：监听响应式数据变化 function watch(source, cb, options = {}) { let oldValue const job = () => { const newValue = effectFn() if (newValue !== oldValue) { cb(newValue, oldValue); oldValue = newValue } } const effectFn = effect(typeof source === 'function' ? source : () => source.value, { lazy: true, scheduler: job }) if (options.immediate) job() else oldValue = effectFn() }

4. Composition API vs Options API 面试高频

Vue2 Options API（按选项组织）
export default { data() { return { count: 0, name: '' } }, computed: { double() { return this.count * 2 } }, methods: { increment() { this.count++ } }, watch: { count(val) { console.log(val) } }, mounted() { /* 生命周期 */ } }
Vue3 Composition API（按逻辑组织）
import { ref, computed, watch, onMounted } from 'vue' export default { setup() { // 逻辑可抽取为 composable const count = ref(0) const double = computed(() => count.value * 2) const increment = () => count.value++ watch(count, val => console.log(val)) onMounted(() => { /* 生命周期 */ }) return { count, double, increment } } }
Composition API 优势： ① 相关逻辑聚合在一起（不像 Options API 分散在 data/methods/computed）； ② 逻辑可复用（抽取为 composable 函数，类似 React Hooks）； ③ 更好的 TypeScript 支持； ④ 树摇（Tree-shaking）友好
