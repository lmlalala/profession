2. Promise 深入 面试必考

// Promise 三种状态：pending → fulfilled / rejected const p = new Promise((resolve, reject) => { // 异步操作... resolve('成功') // 或 reject(new Error('失败')) }) // 链式调用（每个 .then 返回新 Promise） p.then(v => v + '!') .then(v => console.log(v)) // "成功!" .catch(e => console.error(e)) .finally(() => console.log('完成')) // 并发方法 Promise.all([p1, p2, p3]) // 全部成功才 resolve，一个失败即 reject Promise.allSettled([p1, p2, p3]) // 全部完成，收集所有结果 Promise.race([p1, p2, p3]) // 最先 settle 的结果 Promise.any([p1, p2, p3]) // 最先成功的结果（ES2021）

3. async / await 面试必考

// async 函数总是返回 Promise async function fetchUser(id) { const res = await fetch(`/api/users/${id}`) if (!res.ok) throw new Error(`HTTP ${res.status}`) return res.json() } // 错误处理 try { const user = await fetchUser(1) } catch (e) { console.error('请求失败:', e.message) } // 并发执行（不要串行 await！） // ❌ 串行（慢） const u1 = await fetchUser(1) const u2 = await fetchUser(2) // ✅ 并发（快） const [u1, u2] = await Promise.all([fetchUser(1), fetchUser(2)])

4. 并发控制与实用模式

// 限制并发数量（p-limit 原理） async function pLimit(tasks, limit) { const results = [] const executing = new Set() for (const [i, task] of tasks.entries()) { const p = Promise.resolve().then(() => task()) results[i] = p executing.add(p) p.finally(() => executing.delete(p)) if (executing.size >= limit) await Promise.race(executing) } return Promise.all(results) } // 超时控制 function withTimeout(promise, ms) { const timeout = new Promise((_, reject) => setTimeout(() => reject(new Error(`超时 ${ms}ms`)), ms)) return Promise.race([promise, timeout]) } // 重试机制 async function retry(fn, times = 3, delay = 1000) { for (let i = 0; i < times; i++) { try { return await fn() } catch (e) { if (i === times - 1) throw e await new Promise(r => setTimeout(r, delay * (2 ** i))) // 指数退避 } } }
