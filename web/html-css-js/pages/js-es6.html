5. Proxy 与 Reflect 面试高频

Proxy 拦截对象的基本操作（读取、赋值、删除等），是 Vue3 响应式系统的核心。
// 数据验证代理 const validator = { set(target, key, value) { if (key === 'age' && (typeof value !== 'number' || value < 0)) throw new TypeError('age 必须是非负数字') target[key] = value return true } } const person = new Proxy({}, validator) person.age = 25 // ✅ person.age = 'old' // ❌ TypeError // 响应式代理（Vue3 原理简化版） function reactive(obj) { return new Proxy(obj, { get(target, key) { console.log(`读取 ${key}`) return Reflect.get(target, key) }, set(target, key, value) { console.log(`设置 ${key} = ${value}`) return Reflect.set(target, key, value) } }) }
