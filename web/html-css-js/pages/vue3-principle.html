1. effect / track / trigger — 依赖收集核心 面试必考

Vue3 响应式的三层结构：effect.ts 是核心基础（track/trigger/effect），reactive.ts 在其上实现 Proxy 代理，ref.ts 在其上实现基本类型包装。下方代码块带 id，后续 Section 通过 data-import 直接引用，无需重复。
// packages/reactivity/src/effect.ts // 依赖收集核心：track / trigger / effect /** 全局依赖图：WeakMap<target, Map<key, Set<effect>>> */ const targetMap = new WeakMap() let activeEffect = null const effectStack = [] /** * 收集依赖：将当前 activeEffect 添加到 target[key] 的依赖集合中。 * @param {Object} target - 被代理的原始对象 * @param {string|symbol} key - 被访问的属性名 */ function track(target, key) { if (!activeEffect) return let depsMap = targetMap.get(target) if (!depsMap) targetMap.set(target, depsMap = new Map()) let dep = depsMap.get(key) if (!dep) depsMap.set(key, dep = new Set()) dep.add(activeEffect) } /** * 触发更新：执行 target[key] 的所有依赖副作用函数。 * @param {Object} target - 被代理的原始对象 * @param {'ADD'|'SET'|'DELETE'} type - 操作类型 * @param {string|symbol} key - 被修改的属性名 */ function trigger(target, type, key) { const depsMap = targetMap.get(target) if (!depsMap) return const effects = new Set() depsMap.get(key)?.forEach(e => { if (e !== activeEffect) effects.add(e) }) if (type === 'ADD' || type === 'DELETE') depsMap.get('length')?.forEach(e => effects.add(e)) effects.forEach(e => e.options?.scheduler ? e.options.scheduler(e) : e()) } /** * 注册副作用函数，支持 lazy 和 scheduler 选项。 * @param {Function} fn - 副作用函数 * @param {Object} [options={}] - 配置项（lazy/scheduler） * @returns {Function} 包装后的 effectFn */ function effect(fn, options = {}) { const effectFn = () => { effectStack.push(effectFn); activeEffect = effectFn const result = fn() effectStack.pop(); activeEffect = effectStack[effectStack.length - 1] ?? null return result } effectFn.deps = new Set() effectFn.options = options if (!options.lazy) effectFn() return effectFn }
// effect.ts — TypeScript 类型版 type TriggerType = 'ADD' | 'SET' | 'DELETE' interface EffectFn { (): any deps: Set<Set<EffectFn>> options: EffectOptions } interface EffectOptions { lazy?: boolean scheduler?: (e: EffectFn) => void } const targetMap = new WeakMap<object, Map<string | symbol, Set<EffectFn>>>() let activeEffect: EffectFn | null = null const effectStack: EffectFn[] = [] function track(target: object, key: string | symbol): void { if (!activeEffect) return let depsMap = targetMap.get(target) if (!depsMap) targetMap.set(target, depsMap = new Map()) let dep = depsMap.get(key) if (!dep) depsMap.set(key, dep = new Set()) dep.add(activeEffect) } function trigger(target: object, type: TriggerType, key: string | symbol): void { const depsMap = targetMap.get(target) if (!depsMap) return const effects = new Set<EffectFn>() depsMap.get(key)?.forEach(e => { if (e !== activeEffect) effects.add(e) }) if (type === 'ADD' || type === 'DELETE') depsMap.get('length')?.forEach(e => effects.add(e)) effects.forEach(e => e.options?.scheduler ? e.options.scheduler(e) : e()) } function effect(fn: () => any, options: EffectOptions = {}): EffectFn { const effectFn = (() => { effectStack.push(effectFn); activeEffect = effectFn const result = fn() effectStack.pop(); activeEffect = effectStack[effectStack.length - 1] ?? null return result }) as EffectFn effectFn.deps = new Set(); effectFn.options = options if (!options.lazy) effectFn() return effectFn }

2. reactive — Proxy 响应式代理 面试必考

Vue3 用 Proxy 彻底解决了 Vue2 的缺陷：可以拦截属性新增/删除、数组下标赋值、Map/Set 操作，无需递归初始化（懒代理）。
特性Vue2 (defineProperty)Vue3 (Proxy)
属性新增❌ 需 Vue.set()✅ 自动检测
属性删除❌ 需 Vue.delete()✅ 自动检测
数组下标赋值❌ 不响应✅ 自动检测
初始化性能❌ 递归遍历所有属性✅ 懒代理（访问时才代理）
Map/Set 支持❌ 不支持✅ 原生支持
兼容性✅ IE9+❌ 不支持 IE
// packages/reactivity/src/reactive.ts（依赖 effect.js ↑） const reactiveMap = new WeakMap() /** * 将普通对象转为响应式代理，支持深层懒代理。 * @param {Object} raw - 原始对象 * @returns {Proxy} 响应式代理对象 */ function reactive(raw) { if (reactiveMap.has(raw)) return reactiveMap.get(raw) const proxy = new Proxy(raw, { get(target, key, receiver) { track(target, key) const res = Reflect.get(target, key, receiver) return res && typeof res === 'object' ? reactive(res) : res // 懒代理 }, set(target, key, value, receiver) { const hadKey = Object.hasOwn(target, key) const oldVal = target[key] const result = Reflect.set(target, key, value, receiver) if (!hadKey) trigger(target, 'ADD', key) else if (value !== oldVal) trigger(target, 'SET', key) return result }, deleteProperty(target, key) { const hadKey = Object.hasOwn(target, key) const result = Reflect.deleteProperty(target, key) if (hadKey) trigger(target, 'DELETE', key) return result } }) reactiveMap.set(raw, proxy) return proxy } // ── 演示：Vue3 vs Vue2 响应式能力对比 ──────────────────────── const state = reactive({ name: 'Alice', age: 25 }) const logs = [] effect(() => { logs.push(`name=${state.name}`) }) state.name = 'Bob' console.log('修改已有属性 name=Bob → 响应:', logs.join(' | ')) const logs2 = [] effect(() => { logs2.push(`city=${state.city ?? 'undefined'}`) }) state.city = 'Beijing' console.log('✅ 新增属性 city=Beijing → 响应:', logs2.join(' | ')) const logs3 = [] effect(() => { logs3.push(`age=${state.age ?? 'deleted'}`) }) delete state.age console.log('✅ 删除属性 age → 响应:', logs3.join(' | ')) const arr = reactive([1, 2, 3]) const arrLogs = [] effect(() => { arrLogs.push(`arr[0]=${arr[0]}`) }) arr[0] = 99 console.log('✅ 数组下标赋值 arr[0]=99 → 响应:', arrLogs.join(' | ')) const nested = reactive({ a: { b: { c: 1 } } }) const nestedLogs = [] effect(() => { nestedLogs.push(`c=${nested.a.b.c}`) }) nested.a.b.c = 42 console.log('✅ 深层嵌套 nested.a.b.c=42 → 响应:', nestedLogs.join(' | '))
// reactive.ts — TypeScript 类型版 const reactiveMap = new WeakMap<object, object>() function reactive<T extends object>(raw: T): T { if (reactiveMap.has(raw)) return reactiveMap.get(raw) as T const proxy = new Proxy(raw, { get(t, k, r) { track(t, k) const v = Reflect.get(t, k, r) return v && typeof v === 'object' ? reactive(v as object) : v }, set(t, k, v, r) { const had = Object.hasOwn(t, k), old = (t as any)[k] const res = Reflect.set(t, k, v, r) if (!had) trigger(t, 'ADD', k) else if (v !== old) trigger(t, 'SET', k) return res }, deleteProperty(t, k) { const had = Object.hasOwn(t, k) const res = Reflect.deleteProperty(t, k) if (had) trigger(t, 'DELETE', k) return res } }) reactiveMap.set(raw, proxy) return proxy as T }

3. ref / computed / watch 实现

为什么需要 ref？ reactive 只能代理对象，ref 用 getter/setter 包装基本类型（number/string/boolean）。computed 是带缓存的 lazy effect，watch 是带 scheduler 的 lazy effect。
// packages/reactivity/src/ref.ts（依赖 effect.js ↑） /** * 将基本类型包装为响应式引用，通过 .value 读写。 * @param {*} raw - 初始值 * @returns {{ __v_isRef: true, value: * }} */ function ref(raw) { const r = { __v_isRef: true, get value() { track(r, 'value'); return raw }, set value(newVal) { if (newVal !== raw) { raw = newVal; trigger(r, 'SET', 'value') } } } return r } /** * 创建计算属性：依赖变化时标记为"脏"，读取时才重新计算（惰性求值）。 * @param {Function} getter - 计算函数 * @returns {{ __v_isRef: true, value: * }} */ function computed(getter) { let dirty = true, _value const effectFn = effect(getter, { lazy: true, scheduler() { if (!dirty) { dirty = true; trigger(obj, 'SET', 'value') } } }) const obj = { __v_isRef: true, get value() { if (dirty) { _value = effectFn(); dirty = false } track(obj, 'value') return _value } } return obj } /** * 监听 ref 或 getter 函数，变化时执行回调。 * @param {Function|Object} source - 监听源（getter 函数或 ref） * @param {Function} cb - 回调 (newVal, oldVal) * @param {{ immediate?: boolean }} [options={}] */ function watch(source, cb, options = {}) { const getter = typeof source === 'function' ? source : () => source.value let oldValue const job = () => { const newValue = effectFn() if (newValue !== oldValue) { cb(newValue, oldValue); oldValue = newValue } } const effectFn = effect(getter, { lazy: true, scheduler: job }) if (options.immediate) job() else oldValue = effectFn() } // ── 演示 ────────────────────────────────────────────────────── const count = ref(0) const logs = [] effect(() => { logs.push(`count=${count.value}`) }) count.value++; count.value++ console.log('ref count 变化:', logs.join(' → ')) const price = ref(100), quantity = ref(3) let computeCount = 0 const total = computed(() => { computeCount++; return price.value * quantity.value }) console.log('computed total:', total.value, '(计算次数:', computeCount, ')') console.log('再次读取（缓存）:', total.value, '(计算次数:', computeCount, ')') price.value = 200 console.log('price 变化后 total:', total.value, '(计算次数:', computeCount, ')') const name = ref('Alice') const watchLogs = [] watch(name, (newVal, oldVal) => { watchLogs.push(`${oldVal}→${newVal}`) }) name.value = 'Bob'; name.value = 'Charlie' console.log('watch 变化记录:', watchLogs.join(' | '))
// ref.ts — TypeScript 类型版 interface Ref<T> { __v_isRef: true; value: T } interface ComputedRef<T> extends Ref<T> { readonly value: T } function ref<T>(raw: T): Ref<T> { let _v = raw const r: Ref<T> = { __v_isRef: true, get value() { track(r as any, 'value'); return _v }, set value(n: T) { if (n !== _v) { _v = n; trigger(r as any, 'SET', 'value') } } } return r } function computed<T>(getter: () => T): ComputedRef<T> { let dirty = true, _v: T const fn = effect(getter, { lazy: true, scheduler() { if (!dirty) { dirty = true; trigger(obj as any, 'SET', 'value') } } }) const obj: ComputedRef<T> = { __v_isRef: true, get value() { if (dirty) { _v = fn(); dirty = false } track(obj as any, 'value'); return _v } } return obj } type WatchSource<T> = (() => T) | Ref<T> function watch<T>(src: WatchSource<T>, cb: (n: T, o: T) => void, opts: { immediate?: boolean } = {}): void { const getter = typeof src === 'function' ? src as () => T : () => (src as Ref<T>).value let old: T const job = () => { const n = fn() as T; if (n !== old) { cb(n, old); old = n } } const fn = effect(getter, { lazy: true, scheduler: job }) if (opts.immediate) job(); else old = fn() as T }

4. effect 高级选项：lazy / scheduler 面试必考

为什么需要 scheduler？ 默认 effect 每次依赖变化都立即同步执行。scheduler 让你控制"何时执行"——这是 computed 缓存、watch 异步、nextTick 批量更新的共同基础。
// packages/reactivity/src/effect.ts — lazy / scheduler 高级选项演示 // 依赖 effect.js（已通过 import 引入） function reactive(raw) { return new Proxy(raw, { get(t, k, r) { track(t, k); return Reflect.get(t, k, r) }, set(t, k, v, r) { const res = Reflect.set(t, k, v, r); trigger(t, 'SET', k); return res } }) } // ── 演示 1：精确依赖追踪 ────────────────────────────────────── const state = reactive({ a: 1, b: 2, c: 3 }) const e1Logs = [], e2Logs = [] effect(() => { e1Logs.push(state.a + state.b) }) effect(() => { e2Logs.push(state.b + state.c) }) console.log('初始 effect1(a+b):', e1Logs.at(-1), '| effect2(b+c):', e2Logs.at(-1)) state.a = 10 console.log('修改 a=10 → 只触发 effect1:', e1Logs.at(-1), '| effect2 未变:', e2Logs.length === 1) state.b = 20 console.log('修改 b=20 → 两者都触发 effect1:', e1Logs.at(-1), '| effect2:', e2Logs.at(-1)) console.log('依赖图：a→[e1] | b→[e1,e2] | c→[e2]') // ── 演示 2：scheduler 自定义调度（nextTick 批量更新原理）──── console.log('\n--- scheduler: 推迟到微任务执行 ---') const count = reactive({ value: 0 }) const schedulerLog = [] effect(() => { schedulerLog.push(`value=${count.value}`) }, { scheduler(effectFn) { schedulerLog.push('[scheduler] 推迟...') Promise.resolve().then(effectFn) } }) count.value = 1 count.value = 2 console.log('同步阶段:', schedulerLog) await Promise.resolve() console.log('微任务后:', schedulerLog) // ── 演示 3：lazy effect ─────────────────────────────────────── console.log('\n--- lazy: 手动控制执行时机 ---') const price = reactive({ value: 100 }) let computeCount = 0 const lazyFn = effect(() => { computeCount++; return price.value * 2 }, { lazy: true }) console.log('lazy effect 不立即执行，computeCount:', computeCount) const result = lazyFn() console.log('手动执行结果:', result, '| computeCount:', computeCount)
// scheduler 选项的 TypeScript 类型说明 // effect 函数签名中 scheduler 的作用： interface EffectOptions { lazy?: boolean // scheduler：依赖变化时不直接执行 effectFn，而是调用此函数 // 用途：computed（标脏）、watch（异步回调）、nextTick（批量更新） scheduler?: (effectFn: () => any) => void } // computed 内部实现（伪代码） function computed<T>(getter: () => T) { let dirty = true const effectFn = effect(getter, { lazy: true, // 依赖变化时只标脏，不重算 → 惰性求值 scheduler() { dirty = true } }) return { get value() { if (dirty) { dirty = false; return effectFn() } } } } // watch 内部实现（伪代码） function watch<T>(source: () => T, cb: (n: T, o: T) => void) { let oldValue: T const effectFn = effect(source, { lazy: true, // 依赖变化时调用 job（执行回调），而非直接重跑 effect scheduler() { const n = effectFn(); cb(n, oldValue); oldValue = n } }) oldValue = effectFn() }

5. Composition API vs Options API 面试高频

Vue2 Options API（按选项组织）
export default { data() { return { count: 0, name: '' } }, computed: { double() { return this.count * 2 } }, methods: { increment() { this.count++ } }, watch: { count(val) { console.log(val) } }, mounted() { /* 生命周期 */ } }
Vue3 Composition API（按逻辑组织）
import { ref, computed, watch, onMounted } from 'vue' export default { setup() { // 逻辑可抽取为 composable const count = ref(0) const double = computed(() => count.value * 2) const increment = () => count.value++ watch(count, val => console.log(val)) onMounted(() => { /* 生命周期 */ }) return { count, double, increment } } }
Composition API 优势： ① 相关逻辑聚合在一起（不像 Options API 分散在 data/methods/computed）； ② 逻辑可复用（抽取为 composable 函数，类似 React Hooks）； ③ 更好的 TypeScript 支持； ④ 树摇（Tree-shaking）友好

6. Vue3 diff 算法（最长递增子序列）面试必考

Vue3 vs Vue2 diff 核心区别：Vue2 用双端指针；Vue3 在处理乱序节点时，先找出最长递增子序列（LIS），LIS 内的节点无需移动，只移动 LIS 外的节点，将移动次数降到最低。
/** * 求最长递增子序列的下标数组（贪心 + 二分，O(n log n)）。 * 返回的是 LIS 中各元素在原数组中的下标。 * * @param {number[]} arr - 输入数组 * @returns {number[]} LIS 元素的下标列表 */ function getLIS(arr) { const n = arr.length const tails = [] // tails[i] = 长度为 i+1 的递增子序列的最小末尾值 const indices = [] // 记录每个位置对应的原数组下标 const parent = new Array(n).fill(-1) // 用于回溯完整 LIS for (let i = 0; i < n; i++) { const val = arr[i] // 二分查找 tails 中第一个 >= val 的位置 let lo = 0, hi = tails.length while (lo < hi) { const mid = (lo + hi) >> 1 if (tails[mid] < val) lo = mid + 1 else hi = mid } tails[lo] = val indices[lo] = i if (lo > 0) parent[i] = indices[lo - 1] } // 回溯得到完整 LIS 的下标序列 const result = [] let cur = indices[tails.length - 1] while (cur !== -1) { result.unshift(cur); cur = parent[cur] } return result } /** * Vue3 乱序 diff：用 LIS 最小化 DOM 移动次数。 * 返回每个新节点的操作类型（复用/移动/新增）及删除列表。 * * @param {string[]} oldList - 旧子节点 key 列表 * @param {string[]} newList - 新子节点 key 列表 * @returns {{ ops: string[], moves: number, total: number }} 操作摘要 */ function vue3Diff(oldList, newList) { const ops = [] // ① 前缀相同：头部快速跳过 let i = 0 while (i < oldList.length && i < newList.length && oldList[i] === newList[i]) { ops.push(`复用 ${newList[i]}（前缀）`) i++ } // ② 后缀相同：尾部快速跳过 let oldEnd = oldList.length - 1, newEnd = newList.length - 1 while (oldEnd >= i && newEnd >= i && oldList[oldEnd] === newList[newEnd]) { ops.push(`复用 ${newList[newEnd]}（后缀）`) oldEnd--; newEnd-- } // ③ 乱序区间：[i, oldEnd] vs [i, newEnd] const oldKeyMap = new Map(oldList.slice(i, oldEnd + 1).map((k, idx) => [k, i + idx])) // 建立新节点在旧列表中的位置映射（-1 表示新增） const newIndexToOldIndex = [] for (let j = i; j <= newEnd; j++) { const oldIdx = oldKeyMap.get(newList[j]) newIndexToOldIndex.push(oldIdx !== undefined ? oldIdx : -1) } // 找出 LIS（不需要移动的节点集合） const lisIndices = new Set(getLIS(newIndexToOldIndex.filter(v => v !== -1) .map((v, _, a) => v)) ) // 重新映射到 newIndexToOldIndex 的下标 const lisSet = new Set() const sortedOldInNew = [...newIndexToOldIndex].filter(v => v !== -1).sort((a, b) => a - b) const lis = getLIS(newIndexToOldIndex.map(v => v === -1 ? -Infinity : v).filter(v => v !== -Infinity)) lis.forEach(idx => { // idx 是在过滤后数组中的下标，需要映射回 newIndexToOldIndex let cnt = -1 for (let k = 0; k < newIndexToOldIndex.length; k++) { if (newIndexToOldIndex[k] !== -1) cnt++ if (cnt === idx) { lisSet.add(k); break } } }) let moves = 0 for (let j = 0; j < newIndexToOldIndex.length; j++) { const key = newList[i + j] if (newIndexToOldIndex[j] === -1) { ops.push(`新增 ${key}`) } else if (lisSet.has(j)) { ops.push(`复用 ${key}（LIS，无需移动）`) } else { ops.push(`移动 ${key}（LIS 外）`) moves++ } } // ④ 删除旧列表中多余节点 for (let j = i; j <= oldEnd; j++) { if (!newList.includes(oldList[j])) ops.push(`删除 ${oldList[j]}`) } return { ops, moves, total: newList.length } } // 测试 1：Vue3 经典场景 const r1 = vue3Diff(['A','B','C','D','E','F'], ['A','C','E','D','B','F']) console.log('=== 测试1：乱序（Vue3 LIS diff） ===') console.log('旧: A B C D E F → 新: A C E D B F') r1.ops.forEach(op => console.log(' ', op)) console.log(`移动次数: ${r1.moves}（LIS 优化，最小化移动）`) // 测试 2：有新增和删除 const r2 = vue3Diff(['A','B','C','D'], ['B','D','E','F']) console.log('\n=== 测试2：增删 ===') console.log('旧: A B C D → 新: B D E F') r2.ops.forEach(op => console.log(' ', op)) // 对比 Vue2 双端 vs Vue3 LIS 移动次数 console.log('\n=== LIS 核心价值 ===') console.log('旧: 1 2 3 4 5 → 新: 1 3 5 2 4') const r3 = vue3Diff(['1','2','3','4','5'], ['1','3','5','2','4']) console.log('Vue3 LIS 移动次数:', r3.moves, '（只移动不在 LIS 中的节点）')
/** Vue3 diff 操作摘要 */ interface Vue3DiffResult { /** 每一步操作的文字描述 */ ops: string[] /** 实际发生移动的次数 */ moves: number /** 新列表总节点数 */ total: number } /** * 求最长递增子序列的下标数组（贪心 + 二分，O(n log n)）。 * @param arr - 输入数组 * @returns LIS 元素在原数组中的下标列表 */ function getLIS(arr: number[]): number[] { const n = arr.length const tails: number[] = [] const indices: number[] = [] const parent: number[] = new Array(n).fill(-1) for (let i = 0; i < n; i++) { const val = arr[i] let lo = 0, hi = tails.length while (lo < hi) { const mid = (lo + hi) >> 1 if (tails[mid] < val) lo = mid + 1 else hi = mid } tails[lo] = val indices[lo] = i if (lo > 0) parent[i] = indices[lo - 1] } const result: number[] = [] let cur = indices[tails.length - 1] while (cur !== -1) { result.unshift(cur); cur = parent[cur] } return result } /** * Vue3 乱序 diff：用 LIS 最小化 DOM 移动次数。 * @param oldList - 旧子节点 key 列表 * @param newList - 新子节点 key 列表 * @returns 操作摘要 */ function vue3Diff(oldList: string[], newList: string[]): Vue3DiffResult { const ops: string[] = [] let i = 0 while (i < oldList.length && i < newList.length && oldList[i] === newList[i]) { ops.push(`复用 ${newList[i]}（前缀）`); i++ } let oldEnd = oldList.length - 1, newEnd = newList.length - 1 while (oldEnd >= i && newEnd >= i && oldList[oldEnd] === newList[newEnd]) { ops.push(`复用 ${newList[newEnd]}（后缀）`); oldEnd--; newEnd-- } const oldKeyMap = new Map<string, number>( oldList.slice(i, oldEnd + 1).map((k, idx): [string, number] => [k, i + idx]) ) const newIndexToOldIndex: number[] = [] for (let j = i; j <= newEnd; j++) { const oldIdx = oldKeyMap.get(newList[j]) newIndexToOldIndex.push(oldIdx !== undefined ? // ── 响应式核心（来自 effect.js + reactive.js + ref.js）──────── const targetMap = new WeakMap() let activeEffect = null, effectStack = [] function track(target, key) { if (!activeEffect) return let depsMap = targetMap.get(target) if (!depsMap) targetMap.set(target, depsMap = new Map()) let dep = depsMap.get(key) if (!dep) depsMap.set(key, dep = new Set()) dep.add(activeEffect) } function trigger(target, type, key) { const depsMap = targetMap.get(target) if (!depsMap) return const effects = new Set() depsMap.get(key)?.forEach(e => { if (e !== activeEffect) effects.add(e) }) if (type === 'ADD') depsMap.get('length')?.forEach(e => effects.add(e)) effects.forEach(e => e.options?.scheduler ? e.options.scheduler(e) : e()) } function effect(fn, options = {}) { const effectFn = () => { effectStack.push(effectFn); activeEffect = effectFn const result = fn() effectStack.pop(); activeEffect = effectStack[effectStack.length - 1] ?? null return result } effectFn.deps = new Set(); effectFn.options = options if (!options.lazy) effectFn() return effectFn } function reactive(raw) { return new Proxy(raw, { get(t, k, r) { track(t, k); const v = Reflect.get(t, k, r); return v && typeof v === 'object' ? reactive(v) : v }, set(t, k, v, r) { const had = Object.hasOwn(t, k), old = t[k], res = Reflect.set(t, k, v, r); if (!had) trigger(t, 'ADD', k); else if (v !== old) trigger(t, 'SET', k); return res }, deleteProperty(t, k) { const had = Object.hasOwn(t, k), res = Reflect.deleteProperty(t, k); if (had) trigger(t, 'DELETE', k); return res } }) } function ref(raw) { const r = { __v_isRef: true, get value() { track(r, 'value'); return raw }, set value(n) { if (n !== raw) { raw = n; trigger(r, 'SET', 'value') } } } return r } function computed(getter) { let dirty = true, _value const fn = effect(getter, { lazy: true, scheduler() { if (!dirty) { dirty = true; trigger(obj, 'SET', 'value') } } }) const obj = { __v_isRef: true, get value() { if (dirty) { _value = fn(); dirty = false } track(obj, 'value'); return _value } } return obj } function watch(source, cb, options = {}) { const getter = typeof source === 'function' ? source : () => source.value let oldValue const job = () => { const n = fn(); if (n !== oldValue) { cb(n, oldValue); oldValue = n } } const fn = effect(getter, { lazy: true, scheduler: job }) if (options.immediate) job(); else oldValue = fn() } // ── 虚拟 DOM：h / patch ─────────────────────────────────────── function h(type, props, ...children) { return { type, props: props || {}, children: children.flat().filter(c => c != null) } } function createEl(vnode) { if (typeof vnode === 'string' || typeof vnode === 'number') return document.createTextNode(String(vnode)) const el = vnode._el = document.createElement(vnode.type) Object.entries(vnode.props || {}).forEach(([k, v]) => { if (k.startsWith('on') && typeof v === 'function') el.addEventListener(k.slice(2).toLowerCase(), v) else if (k === 'class') el.className = v else if (k === 'style' && typeof v === 'object') Object.assign(el.style, v) else el.setAttribute(k, v) }) vnode.children.forEach(c => el.appendChild(createEl(c))) return el } function patch(container, oldVnode, newVnode) { if (!oldVnode) { container.appendChild(createEl(newVnode)); return } if (oldVnode.type !== newVnode.type) { oldVnode._el.replaceWith(createEl(newVnode)); return } newVnode._el = oldVnode._el newVnode._el.innerHTML = '' newVnode.children.forEach(c => newVnode._el.appendChild(createEl(c))) } // ── createApp：组合响应式 + 虚拟 DOM ───────────────────────── function createApp(component) { return { mount(container) { const ctx = component.setup() let vnode = null effect(() => { const newVnode = component.render(ctx, h) patch(container, vnode, newVnode) vnode = newVnode }) } } } // ── 使用示例 ───────────────────────────────────────────────── const logs = [] const app = createApp({ setup() { const count = ref(0) const double = computed(() => count.value * 2) watch(count, (n, o) => logs.push(`watch: ${o}→${n}`)) return { count, double, inc: () => count.value++, dec: () => count.value-- } }, render({ count, double, inc, dec }, h) { return h('div', null, h('span', null, `count=${count.value} double=${double.value}`), ) } }) // 模拟挂载到虚拟容器（控制台演示） const fakeContainer = { appendChild() {}, innerHTML: '' } app.mount(fakeContainer) console.log('初始 count:', 0) // 手动触发几次更新 const countRef = ref(0) const doubleRef = computed(() => countRef.value * 2) watch(countRef, (n, o) => logs.push(`watch: ${o}→${n}`)) countRef.value = 1 countRef.value = 2 countRef.value = 3 console.log('computed double:', doubleRef.value) console.log('watch 日志:', logs.join(' | ')) console.log('✅ Mini Vue3：reactive + ref + computed + watch + h + patch + createApp 全部工作正常')
