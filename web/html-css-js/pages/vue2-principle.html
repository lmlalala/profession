1. 响应式原理：Object.defineProperty 面试必考

核心思路：遍历 data 的每个属性，用 Object.defineProperty 将其转为 getter/setter。 getter 中收集依赖（Dep.depend），setter 中通知更新（Dep.notify）。
// Dep：依赖收集器（每个响应式属性对应一个 Dep） class Dep { constructor() { this.subs = new Set() } depend() { if (Dep.target) this.subs.add(Dep.target) } notify() { this.subs.forEach(watcher => watcher.update()) } } Dep.target = null // 当前正在计算的 Watcher // Watcher：观察者，封装更新逻辑 class Watcher { constructor(vm, expOrFn, cb) { this.vm = vm; this.cb = cb this.getter = typeof expOrFn === 'function' ? expOrFn : () => vm[expOrFn] this.value = this.get() // 初始化时触发 getter，完成依赖收集 } get() { Dep.target = this // 标记当前 Watcher const value = this.getter() // 触发 getter → dep.depend() Dep.target = null return value } update() { const newVal = this.get() this.cb(newVal, this.value) this.value = newVal } } // observe：将对象属性转为响应式 function defineReactive(obj, key, val) { const dep = new Dep() if (val && typeof val === 'object') observe(val) // 递归 Object.defineProperty(obj, key, { get() { dep.depend(); return val }, set(newVal) { if (newVal === val) return val = newVal if (typeof newVal === 'object') observe(newVal) dep.notify() // 通知所有 Watcher 更新 } }) } function observe(obj) { if (!obj || typeof obj !== 'object') return Object.keys(obj).forEach(key => defineReactive(obj, key, obj[key])) } // 测试响应式系统 const data = { name: 'Alice', age: 25, address: { city: 'Beijing' } } observe(data) // 创建 Watcher，监听 name 变化 new Watcher(data, () => data.name, (newVal, oldVal) => { console.log(`[Watcher] name: "${oldVal}" → "${newVal}"`) }) // 监听嵌套属性 new Watcher(data, () => data.address.city, (newVal, oldVal) => { console.log(`[Watcher] address.city: "${oldVal}" → "${newVal}"`) }) // 触发响应式更新 console.log('初始 name:', data.name) data.name = 'Bob' // 触发 Watcher 回调 data.name = 'Charlie' data.address.city = 'Shanghai' // 嵌套属性也响应 console.log('最终 name:', data.name)

2. Vue2 响应式局限性 面试必考

Object.defineProperty 的缺陷：只能劫持已存在的属性，无法检测属性的新增和删除，也无法检测数组下标直接赋值。
// 响应式基础（复用上面的 Dep/Watcher/observe） class Dep { constructor() { this.subs = new Set() } depend() { if (Dep.target) this.subs.add(Dep.target) } notify() { this.subs.forEach(w => w.update()) } } Dep.target = null class Watcher { constructor(vm, getter, cb) { this.vm = vm; this.cb = cb; this.getter = getter this.value = this.get() } get() { Dep.target = this; const v = this.getter(); Dep.target = null; return v } update() { const n = this.get(); this.cb(n, this.value); this.value = n } } function defineReactive(obj, key, val) { const dep = new Dep() Object.defineProperty(obj, key, { get() { dep.depend(); return val }, set(newVal) { if (newVal !== val) { val = newVal; dep.notify() } } }) } function observe(obj) { if (!obj || typeof obj !== 'object') return Object.keys(obj).forEach(key => defineReactive(obj, key, obj[key])) } const data = { name: 'Alice', list: [1, 2, 3] } observe(data) const changes = [] new Watcher(data, () => data.name, (n, o) => changes.push(`name: ${o}→${n}`)) // ✅ 正常响应：修改已有属性 data.name = 'Bob' console.log('✅ 修改已有属性 name=Bob → 响应:', changes) // ❌ 无法检测：新增属性 const before = changes.length data.newProp = 'new' console.log('❌ 新增属性 newProp → 无响应（changes 未增加）:', changes.length === before) // ❌ 无法检测：删除属性 const before2 = changes.length delete data.name console.log('❌ 删除属性 name → 无响应:', changes.length === before2) // ❌ 数组下标赋值无法检测 console.log('❌ data.list[0] = 99 → 无响应（Vue2 缺陷）') data.list[0] = 99 console.log('list:', JSON.stringify(data.list)) console.log('\n💡 Vue2 解决方案：Vue.set(obj, key, val) / vm.$set') console.log('💡 Vue3 用 Proxy 彻底解决了这些问题')

3. 虚拟 DOM（Virtual DOM）面试必考

核心思想：用 JS 对象描述 DOM 结构（VNode），数据变化时先对比新旧 VNode（diff），再将最小差异更新到真实 DOM。
// VNode 结构：用 JS 对象描述 DOM function h(tag, props, ...children) { return { tag, props: props || {}, children: children.flat().filter(c => c != null) } } // 示例 VNode 树 const vnode = h('div', { id: 'app', class: 'container' }, h('h1', null, 'Hello Vue'), h('ul', null, h('li', { key: 'a' }, 'Item A'), h('li', { key: 'b' }, 'Item B'), h('li', { key: 'c' }, 'Item C'), ) ) // 打印 VNode 结构 function printVNode(node, depth = 0) { const indent = ' '.repeat(depth) if (typeof node === 'string') { console.log(`${indent}"${node}"`); return } const props = Object.entries(node.props).map(([k,v]) => `${k}="${v}"`).join(' ') console.log(`${indent}<${node.tag}${props ? ' ' + props : ''}>`) node.children.forEach(c => printVNode(c, depth + 1)) console.log(`${indent}</${node.tag}>`) } console.log('VNode 树结构：') printVNode(vnode) // VNode 转 JSON（数据结构） console.log('\nVNode JSON（简化）:') function vnodeToSimple(node) { if (typeof node === 'string') return node return { tag: node.tag, props: node.props, children: node.children.map(vnodeToSimple) } } console.log(JSON.stringify(vnodeToSimple(vnode), null, 2))

4. diff 算法（同层对比）面试必考

Vue2 diff 策略：同层比较，不跨层。使用双端指针（头头/尾尾/头尾/尾头）对比，再用 key 建立映射表。 时间复杂度 O(n)。
// diff 算法：分析新旧列表的差异 function diffLists(oldList, newList) { const oldMap = new Map(oldList.map((v, i) => [v, i])) const newMap = new Map(newList.map((v, i) => [v, i])) const removed = oldList.filter(v => !newMap.has(v)) const added = newList.filter(v => !oldMap.has(v)) const moved = newList.filter(v => oldMap.has(v) && newMap.get(v) !== oldMap.get(v)) const same = newList.filter(v => oldMap.has(v) && newMap.get(v) === oldMap.get(v)) return { removed, added, moved, same } } // 测试 1：基本 diff const old1 = ['A', 'B', 'C', 'D', 'E'] const new1 = ['A', 'C', 'E', 'B', 'D'] const result1 = diffLists(old1, new1) console.log('旧列表:', old1.join(', ')) console.log('新列表:', new1.join(', ')) console.log('复用（位置不变）:', result1.same.join(', ') || '无') console.log('移动（key相同位置变）:', result1.moved.join(', ') || '无') console.log('删除:', result1.removed.join(', ') || '无') console.log('新增:', result1.added.join(', ') || '无') // 测试 2：有新增和删除 const old2 = ['A', 'B', 'C'] const new2 = ['B', 'C', 'D', 'E'] const result2 = diffLists(old2, new2) console.log('\n旧列表:', old2.join(', ')) console.log('新列表:', new2.join(', ')) console.log('删除:', result2.removed.join(', ')) console.log('新增:', result2.added.join(', ')) // 双端指针策略说明 console.log('\n双端指针策略（Vue2 核心）：') console.log('oldStart↔newStart oldEnd↔newEnd') console.log('oldStart↔newEnd oldEnd↔newStart') console.log('有 key：建立 key→index 映射，O(1) 查找复用节点')

5. Mini Vue2 完整实现

// Mini Vue2：响应式 + computed + watch + 渲染 class Dep { constructor() { this.subs = new Set() } depend() { if (Dep.target) this.subs.add(Dep.target) } notify() { this.subs.forEach(w => w.update()) } } Dep.target = null class Watcher { constructor(vm, getter, cb, lazy = false) { this.vm = vm; this.cb = cb; this.getter = getter; this.lazy = lazy; this.dirty = lazy this.value = lazy ? undefined : this.get() } get() { Dep.target = this; const v = this.getter.call(this.vm); Dep.target = null; return v } update() { if (this.lazy) { this.dirty = true } else { const n = this.get(); this.cb(n, this.value); this.value = n } } evaluate() { this.value = this.get(); this.dirty = false } } function defineReactive(obj, key, val) { const dep = new Dep() if (val && typeof val === 'object') observe(val) Object.defineProperty(obj, key, { get() { dep.depend(); return val }, set(newVal) { if (newVal !== val) { val = newVal; dep.notify() } } }) } function observe(obj) { if (!obj || typeof obj !== 'object') return Object.keys(obj).forEach(key => defineReactive(obj, key, obj[key])) } // Mini Vue2 核心 class MiniVue { constructor(options) { this.$data = typeof options.data === 'function' ? options.data() : options.data observe(this.$data) // proxy：this.name → this.$data.name Object.keys(this.$data).forEach(key => { Object.defineProperty(this, key, { get() { return this.$data[key] }, set(v) { this.$data[key] = v } }) }) // computed if (options.computed) { Object.keys(options.computed).forEach(key => { const watcher = new Watcher(this, options.computed[key], () => {}, true) Object.defineProperty(this, key, { get() { if (watcher.dirty) watcher.evaluate(); return watcher.value } }) }) } // watch if (options.watch) { Object.keys(options.watch).forEach(key => { new Watcher(this, () => this.$data[key], options.watch[key].bind(this)) }) } // methods if (options.methods) { Object.keys(options.methods).forEach(k => { this[k] = options.methods[k].bind(this) }) } options.created?.call(this) } } // 测试 Mini Vue2 const vm = new MiniVue({ data() { return { count: 0, name: 'Alice' } }, computed: { doubleCount() { return this.count * 2 }, greeting() { return `Hello, ${this.name}!` } }, watch: { count(newVal, oldVal) { console.log(`[watch] count: ${oldVal} → ${newVal}`) } }, methods: { increment() { this.count++ }, setName(name) { this.name = name } }, created() { console.log('[created] Mini Vue2 实例已创建') } }) console.log('初始 count:', vm.count) console.log('computed doubleCount:', vm.doubleCount) console.log('computed greeting:', vm.greeting) vm.increment() vm.increment() console.log('count 自增2次后:', vm.count) console.log('doubleCount:', vm.doubleCount) vm.setName('Bob') console.log('修改 name 后 greeting:', vm.greeting)
