2. Fiber 架构 面试必考

为什么需要 Fiber？ React 15 的 Stack Reconciler 是同步递归的，一旦开始无法中断，大型组件树会阻塞主线程导致卡顿。 React 16 引入 Fiber，将渲染工作拆分为可中断的小单元，利用浏览器空闲时间（requestIdleCallback）执行。

3. React diff 算法 面试必考

React diff 三大策略： ① 同层比较（不跨层）； ② 类型不同直接替换； ③ key 相同则复用（key 的重要性！）
/** * React 多节点 diff（lastPlacedIndex 算法，简化版）。 * 策略：两轮遍历 * 第一轮：顺序扫描，key 相同则复用，key 不同立即退出 * 第二轮：剩余旧节点建立 key→index 映射表，遍历剩余新节点 * 用 lastPlacedIndex 判断是否需要移动 * * @param {string[]} oldList - 旧子节点 key 列表 * @param {string[]} newList - 新子节点 key 列表 * @returns {{ ops: string[], moves: number }} 操作日志与移动次数 */ function reactDiff(oldList, newList) { const ops = [] let lastPlacedIndex = 0 // 最后一个「不需要移动」的旧节点下标 let newIdx = 0 let oldIdx = 0 // ── 第一轮：顺序遍历，处理前缀 key 相同的节点 ── while (oldIdx < oldList.length && newIdx < newList.length) { if (oldList[oldIdx] !== newList[newIdx]) break // key 不同，退出第一轮 ops.push(`复用 ${newList[newIdx]}（第一轮，index=${oldIdx}）`) lastPlacedIndex = oldIdx oldIdx++; newIdx++ } // ── 第二轮：剩余旧节点建立 key→oldIndex 映射表 ── const existingMap = new Map() for (let i = oldIdx; i < oldList.length; i++) { existingMap.set(oldList[i], i) } let moves = 0 // 遍历剩余新节点 for (; newIdx < newList.length; newIdx++) { const key = newList[newIdx] const oldIndex = existingMap.get(key) if (oldIndex === undefined) { // 旧列表中不存在：新增 ops.push(`新增 ${key}`) } else { existingMap.delete(key) if (oldIndex < lastPlacedIndex) { // 旧位置在 lastPlacedIndex 之前：需要向后移动 ops.push(`移动 ${key}（oldIndex=${oldIndex} < lastPlacedIndex=${lastPlacedIndex}）`) moves++ } else { // 旧位置在 lastPlacedIndex 之后或相等：不需要移动，原地复用 ops.push(`复用 ${key}（第二轮，oldIndex=${oldIndex}，无需移动）`) lastPlacedIndex = oldIndex } } } // 剩余旧节点全部删除 existingMap.forEach((_, key) => ops.push(`删除 ${key}`)) return { ops, moves } } // 测试 1：React 经典场景（lastPlacedIndex 的体现） const r1 = reactDiff(['A','B','C','D'], ['D','A','B','C']) console.log('=== 测试1：D 移到最前 ===') console.log('旧: A B C D → 新: D A B C') r1.ops.forEach(op => console.log(' ', op)) console.log(`移动次数: ${r1.moves}（React 会移动 A B C，而非只移动 D！）`) // 测试 2：有新增和删除 const r2 = reactDiff(['A','B','C','D','E'], ['D','A','C','B','F']) console.log('\n=== 测试2：综合场景 ===') console.log('旧: A B C D E → 新: D A C B F') r2.ops.forEach(op => console.log(' ', op)) console.log(`移动次数: ${r2.moves}`) // 对比：Vue2 双端 vs React lastPlacedIndex console.log('\n=== 关键区别 ===') console.log('React 不用双端指针，只从左到右扫描一遍') console.log('lastPlacedIndex 保证：旧位置在其右侧的节点不需要移动') console.log('缺点：[A,B,C,D]→[D,A,B,C] React 移动 A/B/C，Vue2 只移动 D')

4. Hooks 原理实现 面试必考

Hooks 本质：每个组件对应一个 Fiber，Fiber 上有一个 memoizedState 链表，每个 Hook 调用对应链表中的一个节点。 这就是为什么 Hooks 不能在条件语句中调用——顺序必须固定！
let currentFiber = null let hookIndex = 0 /** * 创建模拟 Fiber 节点，用于挂载 Hooks 链表。 * * @returns {{ hooks: Array, renderCount: number }} Fiber 对象 */ function createFiber() { return { hooks: [], renderCount: 0 } } /** * 模拟 React.useState，基于 Fiber hooks 链表实现状态管理。 * * @param {*} initial - 初始状态值或惰性初始化函数 * @returns {[*, Function]} [当前状态, setState 函数] */ function useState(initial) { const fiber = currentFiber const idx = hookIndex++ if (fiber.hooks[idx] === undefined) { fiber.hooks[idx] = { state: typeof initial === 'function' ? initial() : initial, queue: [] } } const hook = fiber.hooks[idx] while (hook.queue.length) { const action = hook.queue.shift() hook.state = typeof action === 'function' ? action(hook.state) : action } const setState = (action) => { hook.queue.push(action); render() } return [hook.state, setState] } /** * 模拟 React.useReducer，通过 reducer 函数管理复杂状态。 * * @param {Function} reducer - 状态归约函数 (state, action) => newState * @param {*} initial - 初始状态 * @returns {[*, Function]} [当前状态, dispatch 函数] */ function useReducer(reducer, initial) { const fiber = currentFiber const idx = hookIndex++ if (fiber.hooks[idx] === undefined) fiber.hooks[idx] = { state: initial, queue: [] } const hook = fiber.hooks[idx] while (hook.queue.length) hook.state = reducer(hook.state, hook.queue.shift()) const dispatch = (action) => { hook.queue.push(action); render() } return [hook.state, dispatch] } /** * 模拟 React.useMemo，依赖未变化时返回缓存值。 * * @param {Function} factory - 计算函数 * @param {Array} deps - 依赖数组 * @returns {*} 缓存或重新计算的值 */ function useMemo(factory, deps) { const fiber = currentFiber const idx = hookIndex++ const prev = fiber.hooks[idx] const hasChanged = !prev || deps.some((d, i) => d !== prev.deps[i]) if (hasChanged) { const value = factory() fiber.hooks[idx] = { value, deps } return value } return prev.value } const fiber = createFiber() const renderLog = [] function render() { currentFiber = fiber; hookIndex = 0; fiber.renderCount++ const [count, setCount] = useState(0) const [name, setName] = useState('Alice') const [state, dispatch] = useReducer((s, a) => { if (a.type === 'INC') return { ...s, n: s.n + 1 } if (a.type === 'DEC') return { ...s, n: s.n - 1 } return s }, { n: 10 }) const doubled = useMemo(() => count * 2, [count]) renderLog.push({ count, name, reducerN: state.n, doubled }) return { count, name, state, doubled, setCount, setName, dispatch } } let result = render() console.log('初始渲染:', JSON.stringify(renderLog.at(-1))) result.setCount(1); result = render() console.log('setCount(1):', JSON.stringify(renderLog.at(-1))) result.setCount(c => c + 1); result = render() console.log('setCount(c=>c+1):', JSON.stringify(renderLog.at(-1))) result.dispatch({ type: 'INC' }); result.dispatch({ type: 'INC' }); result = render() console.log('dispatch INC×2:', JSON.stringify(renderLog.at(-1))) let memoCalcCount = 0 const fiber2 = createFiber() function renderWithMemo(count) { currentFiber = fiber2; hookIndex = 0 const [c] = useState(count) const expensive = useMemo(() => { memoCalcCount++; return c * 100 }, [c]) return expensive } renderWithMemo(5); renderWithMemo(5); renderWithMemo(5) console.log('useMemo 相同deps，计算次数:', memoCalcCount, '(应为1，后两次缓存)') renderWithMemo(6) console.log('useMemo deps变化，计算次数:', memoCalcCount, '(应为2)') console.log('总渲染次数:', fiber.renderCount)
