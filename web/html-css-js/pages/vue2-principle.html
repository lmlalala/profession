1. 响应式原理：Object.defineProperty 面试必考

核心思路：遍历 data 的每个属性，用 Object.defineProperty 将其转为 getter/setter。 getter 中收集依赖（Dep.depend），setter 中通知更新（Dep.notify）。
// Dep：依赖收集器（每个响应式属性对应一个 Dep） class Dep { constructor() { this.subs = [] } depend() { if (Dep.target) this.subs.push(Dep.target) } notify() { this.subs.forEach(watcher => watcher.update()) } } Dep.target = null // 当前正在计算的 Watcher // Watcher：观察者，封装更新逻辑 class Watcher { constructor(vm, expOrFn, cb) { this.vm = vm; this.cb = cb this.getter = typeof expOrFn === 'function' ? expOrFn : () => vm[expOrFn] this.value = this.get() // 初始化时触发 getter，完成依赖收集 } get() { Dep.target = this // 标记当前 Watcher const value = this.getter() // 触发 getter → dep.depend() Dep.target = null return value } update() { const newVal = this.get() this.cb(newVal, this.value) this.value = newVal } } // observe：将对象属性转为响应式 function defineReactive(obj, key, val) { const dep = new Dep() if (val && typeof val === 'object') observe(val) // 递归 Object.defineProperty(obj, key, { get() { dep.depend(); return val }, set(newVal) { if (newVal === val) return val = newVal if (typeof newVal === 'object') observe(newVal) dep.notify() // 通知所有 Watcher 更新 } }) }

2. Vue2 响应式局限性 面试必考

Object.defineProperty 的缺陷：只能劫持已存在的属性，无法检测属性的新增和删除，也无法检测数组下标直接赋值。
// ❌ 无法检测的操作 vm.data.newProp = 'new' // 新增属性 → 不响应 delete vm.data.name // 删除属性 → 不响应 vm.list[0] = 'changed' // 数组下标赋值 → 不响应 vm.list.length = 0 // 修改 length → 不响应 // ✅ Vue2 的解决方案 Vue.set(vm.data, 'newProp', 'new') // 新增属性 vm.$set(vm.data, 'newProp', 'new') // 同上 Vue.delete(vm.data, 'name') // 删除属性 // 数组：Vue2 重写了 7 个变异方法 // push/pop/shift/unshift/splice/sort/reverse vm.list.push('new item') // ✅ 响应式 vm.list.splice(0, 1, 'changed') // ✅ 响应式

3. 虚拟 DOM（Virtual DOM）面试必考

核心思想：用 JS 对象描述 DOM 结构（VNode），数据变化时先对比新旧 VNode（diff），再将最小差异更新到真实 DOM。
// VNode 结构 function h(tag, props, ...children) { return { tag, props: props || {}, children: children.flat() } } // 示例 VNode 树 const vnode = h('div', { id: 'app', class: 'container' }, h('h1', null, 'Hello Vue'), h('ul', null, h('li', { key: 'a' }, 'Item A'), h('li', { key: 'b' }, 'Item B'), ) ) // mount：VNode → 真实 DOM function mount(vnode, container) { if (typeof vnode === 'string') { container.appendChild(document.createTextNode(vnode)) return } const el = vnode._el = document.createElement(vnode.tag) Object.entries(vnode.props).forEach(([k, v]) => el.setAttribute(k, v)) vnode.children.forEach(child => mount(child, el)) container.appendChild(el) }

4. diff 算法（同层对比）面试必考

Vue2 diff 策略：同层比较，不跨层。使用双端指针（头头/尾尾/头尾/尾头）对比，再用 key 建立映射表。 时间复杂度 O(n)。
// 简化版 patch（同层 diff） function patch(oldVnode, newVnode) { // tag 不同 → 直接替换 if (oldVnode.tag !== newVnode.tag) { oldVnode._el.replaceWith(createEl(newVnode)); return } // 更新 props patchProps(oldVnode._el, oldVnode.props, newVnode.props) // 对比子节点（核心 diff） patchChildren(oldVnode._el, oldVnode.children, newVnode.children) } // 有 key：建立 key→index 映射，O(1) 查找复用节点 // 无 key：双端指针逐一比较 // 双端指针策略（Vue2 核心）： // oldStart↔newStart oldEnd↔newEnd // oldStart↔newEnd oldEnd↔newStart

5. Mini Vue2 完整实现

// Mini Vue2：响应式 + 模板编译 + 虚拟 DOM class MiniVue { constructor(options) { this.$options = options this.$data = options.data() this.$el = document.querySelector(options.el) observe(this.$data) proxy(this, this.$data) // this.name → this.$data.name if (options.methods) bindMethods(this, options.methods) if (options.computed) initComputed(this, options.computed) if (options.watch) initWatch(this, options.watch) options.created?.call(this) this._update() // 初次渲染 } _update() { const vnode = this.$options.render.call(this, h) if (this._vnode) patch(this._vnode, vnode) else mount(vnode, this.$el) this._vnode = vnode } }
