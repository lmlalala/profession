3. async / await 面试必考

/** * 延迟指定毫秒后 resolve 一个值。 * * @param {number} ms - 延迟毫秒数 * @param {string} val - resolve 的值 * @returns {Promise<string>} 延迟后 resolve 的 Promise */ const delay = (ms, val) => new Promise(r => setTimeout(() => r(val), ms)) // 串行 vs 并发对比 console.log('--- 串行执行（慢）---') const t1 = Date.now() const r1 = await delay(50, 'user1') const r2 = await delay(50, 'user2') console.log(`串行耗时：~${Date.now() - t1}ms，结果：${r1} ${r2}`) console.log('--- 并发执行（快）---') const t2 = Date.now() const [u1, u2] = await Promise.all([ delay(50, 'user1'), delay(50, 'user2') ]) console.log(`并发耗时：~${Date.now() - t2}ms，结果：${u1} ${u2}`) /** * 模拟可能失败的异步操作。 * * @param {boolean} shouldFail - 是否模拟失败 * @returns {Promise<string>} 操作结果 * @throws {Error} 当 shouldFail 为 true 时抛出 */ async function riskyOp(shouldFail) { if (shouldFail) throw new Error('操作失败') return '操作成功' } try { console.log(await riskyOp(false)) console.log(await riskyOp(true)) } catch(e) { console.error('try/catch 捕获:', e.message) } // async 函数总是返回 Promise const result = await (async () => 42)() console.log('async 函数返回 Promise，await 解包:', result)

4. 并发控制与实用模式

/** * 延迟指定毫秒后 resolve 一个值。 * * @param {number} ms - 延迟毫秒数 * @param {string} val - resolve 的值 * @returns {Promise<string>} 延迟后 resolve 的 Promise */ const delay = (ms, val) => new Promise(r => setTimeout(() => r(val), ms)) /** * 限制并发数量执行异步任务（p-limit 原理）。 * * @param {Array<() => Promise<*>>} tasks - 返回 Promise 的任务函数数组 * @param {number} limit - 最大并发数 * @returns {Promise<Array<*>>} 所有任务的结果数组（保持顺序） */ async function pLimit(tasks, limit) { const results = [] const executing = new Set() for (const [i, task] of tasks.entries()) { const p = Promise.resolve().then(() => task()) results[i] = p executing.add(p) p.finally(() => executing.delete(p)) if (executing.size >= limit) await Promise.race(executing) } return Promise.all(results) } const tasks = Array.from({ length: 6 }, (_, i) => () => delay(30, `task${i + 1}`)) const t = Date.now() const res = await pLimit(tasks, 2) console.log('并发限制 2，6 个任务结果:', res.join(', ')) console.log(`耗时约：${Date.now() - t}ms（理论 ~90ms）`) /** * 为 Promise 添加超时控制，超时则 reject。 * * @param {Promise<*>} promise - 要控制超时的 Promise * @param {number} ms - 超时毫秒数 * @returns {Promise<*>} 原 Promise 或超时 reject */ function withTimeout(promise, ms) { const timeout = new Promise((_, reject) => setTimeout(() => reject(new Error(`超时 ${ms}ms`)), ms) ) return Promise.race([promise, timeout]) } try { await withTimeout(delay(200, 'ok'), 50) } catch(e) { console.warn('超时控制:', e.message) } const ok = await withTimeout(delay(10, '成功'), 100) console.log('超时控制（未超时）:', ok) /** * 重试机制 —— 失败后自动重试指定次数。 * * @param {() => Promise<*>} fn - 要重试的异步函数 * @param {number} times - 最大重试次数 * @returns {Promise<*>} 成功的结果 * @throws {Error} 全部重试失败后抛出最后一次错误 */ let attempt = 0 async function retry(fn, times = 3) { for (let i = 0; i < times; i++) { try { return await fn() } catch(e) { if (i === times - 1) throw e console.warn(`第${i + 1}次失败，重试...`) } } } const result = await retry(async () => { attempt++ if (attempt < 3) throw new Error('暂时失败') return `第${attempt}次成功` }) console.log('重试机制:', result)
const delay = (ms: number, val: string): Promise<string> => new Promise(r => setTimeout(() => r(val), ms)) type Task<T> = () => Promise<T> async function pLimit<T>(tasks: Task<T>[], limit: number): Promise<T[]> { const results: Promise<T>[] = [] const executing = new Set<Promise<T>>() for (const [i, task] of tasks.entries()) { const p: Promise<T> = Promise.resolve().then(() => task()) results[i] = p executing.add(p) p.finally(() => executing.delete(p)) if (executing.size >= limit) await Promise.race(executing) } return Promise.all(results) } const tasks: Task<string>[] = Array.from( { length: 6 }, (_, i) => () => delay(30, `task${i + 1}`) ) const t: number = Date.now() const res: string[] = await pLimit(tasks, 2) console.log('并发限制 2，6 个任务结果:', res.join(', ')) console.log(`耗时约：${Date.now() - t}ms（理论 ~90ms）`) function withTimeout<T>(promise: Promise<T>, ms: number): Promise<T> { const timeout = new Promise<never>((_, reject) => setTimeout(() => reject(new Error(`超时 ${ms}ms`)), ms) ) return Promise.race([promise, timeout]) } try { await withTimeout(delay(200, 'ok'), 50) } catch (e) { console.warn('超时控制:', (e as Error).message) } const ok: string = await withTimeout(delay(10, '成功'), 100) console.log('超时控制（未超时）:', ok) let attempt = 0 async function retry<T>(fn: () => Promise<T>, times: number = 3): Promise<T> { for (let i = 0; i < times; i++) { try { return await fn() } catch (e) { if (i === times - 1) throw e console.warn(`第${i + 1}次失败，重试...`) } } throw new Error('unreachable') } const result: string = await retry(async () => { attempt++ if (attempt < 3) throw new Error('暂时失败') return `第${attempt}次成功` }) console.log('重试机制:', result)
