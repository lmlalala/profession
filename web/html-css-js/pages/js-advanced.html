手写深拷贝 面试必考

function deepClone(value, map = new WeakMap()) { // 处理 null 和基本类型 if (value === null || typeof value !== 'object') return value // 处理特殊对象 if (value instanceof Date) return new Date(value) if (value instanceof RegExp) return new RegExp(value) if (value instanceof Map) return new Map([...value].map(([k,v]) => [k, deepClone(v, map)])) if (value instanceof Set) return new Set([...value].map(v => deepClone(v, map))) // 处理循环引用 if (map.has(value)) return map.get(value) const clone = Array.isArray(value) ? [] : Object.create(Object.getPrototypeOf(value)) map.set(value, clone) for (const key of Reflect.ownKeys(value)) { clone[key] = deepClone(value[key], map) } return clone }

2. 防抖（debounce）与节流（throttle）面试必考

防抖：最后一次触发后等待 n ms 才执行（搜索框输入）
节流：每 n ms 最多执行一次（滚动、resize）
// 防抖（支持立即执行） function debounce(fn, delay, immediate = false) { let timer = null return function(...args) { if (timer) clearTimeout(timer) if (immediate && !timer) fn.apply(this, args) timer = setTimeout(() => { if (!immediate) fn.apply(this, args) timer = null }, delay) } } // 节流（时间戳版） function throttle(fn, interval) { let last = 0 return function(...args) { const now = Date.now() if (now - last >= interval) { last = now return fn.apply(this, args) } } }

// 纯函数：相同输入 → 相同输出，无副作用 const add = (a, b) => a + b // ✅ 纯函数 // 柯里化（Currying） function curry(fn) { return function curried(...args) { if (args.length >= fn.length) return fn.apply(this, args) return (...more) => curried(...args, ...more) } } const curriedAdd = curry((a, b, c) => a + b + c) curriedAdd(1)(2)(3) // 6 curriedAdd(1, 2)(3) // 6 // 函数组合（compose / pipe） const compose = (...fns) => x => fns.reduceRight((v, f) => f(v), x) const pipe = (...fns) => x => fns.reduce((v, f) => f(v), x) const double = x => x * 2 const addOne = x => x + 1 const square = x => x * x const transform = pipe(double, addOne, square) transform(3) // pipe: (3*2+1)^2 = 49
