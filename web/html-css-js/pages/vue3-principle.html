1. Proxy 响应式 vs Vue2 Object.defineProperty 面试必考

Vue3 用 Proxy 彻底解决了 Vue2 的缺陷：可以拦截属性新增/删除、数组下标赋值、Map/Set 操作，无需递归初始化（懒代理）。
特性Vue2 (defineProperty)Vue3 (Proxy)
属性新增❌ 需 Vue.set()✅ 自动检测
属性删除❌ 需 Vue.delete()✅ 自动检测
数组下标赋值❌ 不响应✅ 自动检测
初始化性能❌ 递归遍历所有属性✅ 懒代理（访问时才代理）
Map/Set 支持❌ 不支持✅ 原生支持
兼容性✅ IE9+❌ 不支持 IE
// Vue3 响应式核心：Proxy + Reflect + track/trigger const targetMap = new WeakMap() let activeEffect = null const effectStack = [] /** * 收集依赖：将当前 activeEffect 添加到 target[key] 的依赖集合中。 * @param {Object} target - 被代理的原始对象 * @param {string|symbol} key - 被访问的属性名 * @returns {void} */ function track(target, key) { if (!activeEffect) return let depsMap = targetMap.get(target) if (!depsMap) targetMap.set(target, depsMap = new Map()) let dep = depsMap.get(key) if (!dep) depsMap.set(key, dep = new Set()) dep.add(activeEffect) } /** * 触发更新：执行 target[key] 的所有依赖副作用函数。 * @param {Object} target - 被代理的原始对象 * @param {'ADD'|'SET'|'DELETE'} type - 操作类型 * @param {string|symbol} key - 被修改的属性名 * @returns {void} */ function trigger(target, type, key) { const depsMap = targetMap.get(target) if (!depsMap) return const effects = new Set() depsMap.get(key)?.forEach(e => { if (e !== activeEffect) effects.add(e) }) if (type === 'ADD' || type === 'DELETE') depsMap.get('length')?.forEach(e => effects.add(e)) effects.forEach(e => e()) } /** * 注册副作用函数，立即执行一次以收集依赖。 * @param {Function} fn - 副作用函数 * @returns {Function} 包装后的 effectFn */ function effect(fn) { const effectFn = () => { effectStack.push(effectFn); activeEffect = effectFn const result = fn() effectStack.pop(); activeEffect = effectStack[effectStack.length - 1] ?? null return result } effectFn.deps = new Set() effectFn() return effectFn } const reactiveMap = new WeakMap() /** * 将普通对象转为响应式代理，支持深层懒代理。 * @param {Object} raw - 原始对象 * @returns {Proxy} 响应式代理对象 */ function reactive(raw) { if (reactiveMap.has(raw)) return reactiveMap.get(raw) const proxy = new Proxy(raw, { get(target, key, receiver) { track(target, key) const res = Reflect.get(target, key, receiver) return res && typeof res === 'object' ? reactive(res) : res }, set(target, key, value, receiver) { const hadKey = Object.hasOwn(target, key) const oldVal = target[key] const result = Reflect.set(target, key, value, receiver) if (!hadKey) trigger(target, 'ADD', key) else if (value !== oldVal) trigger(target, 'SET', key) return result }, deleteProperty(target, key) { const hadKey = Object.hasOwn(target, key) const result = Reflect.deleteProperty(target, key) if (hadKey) trigger(target, 'DELETE', key) return result } }) reactiveMap.set(raw, proxy) return proxy } // ── 演示 ── const state = reactive({ name: 'Alice', age: 25 }) const logs = [] effect(() => { logs.push(`name=${state.name}`) }) state.name = 'Bob' console.log('修改已有属性 name=Bob → 响应:', logs.join(' | ')) // ✅ 新增属性（Vue2 无法检测，Vue3 可以） const logs2 = [] effect(() => { logs2.push(`city=${state.city ?? 'undefined'}`) }) state.city = 'Beijing' console.log('✅ 新增属性 city=Beijing → 响应:', logs2.join(' | ')) // ✅ 删除属性 const logs3 = [] effect(() => { logs3.push(`age=${state.age ?? 'deleted'}`) }) delete state.age console.log('✅ 删除属性 age → 响应:', logs3.join(' | ')) // ✅ 数组下标赋值 const arr = reactive([1, 2, 3]) const arrLogs = [] effect(() => { arrLogs.push(`arr[0]=${arr[0]}`) }) arr[0] = 99 console.log('✅ 数组下标赋值 arr[0]=99 → 响应:', arrLogs.join(' | ')) // ✅ 深层嵌套（懒代理） const nested = reactive({ a: { b: { c: 1 } } }) const nestedLogs = [] effect(() => { nestedLogs.push(`c=${nested.a.b.c}`) }) nested.a.b.c = 42 console.log('✅ 深层嵌套 nested.a.b.c=42 → 响应:', nestedLogs.join(' | '))

e.options.scheduler(e) : e()) } /** * 注册副作用函数，支持 lazy 和 scheduler 选项。 * @param {Function} fn - 副作用函数 * @param {Object} [options={}] - 配置项 * @param {boolean} [options.lazy] - 为 true 时不立即执行 * @param {Function} [options.scheduler] - 自定义调度器 * @returns {Function} 包装后的 effectFn */ function effect(fn, options = {}) { const effectFn = () => { effectStack.push(effectFn); activeEffect = effectFn const result = fn() effectStack.pop(); activeEffect = effectStack[effectStack.length - 1] ?? null return result } effectFn.deps = new Set() effectFn.options = options if (!options.lazy) effectFn() return effectFn } /** * 创建响应式代理（简化版，仅拦截 get/set）。 * @param {Object} raw - 原始对象 * @returns {Proxy} 响应式代理对象 */ function reactive(raw) { return new Proxy(raw, { get(target, key, receiver) { track(target, key) return Reflect.get(target, key, receiver) }, set(target, key, value, receiver) { const result = Reflect.set(target, key, value, receiver) trigger(target, 'SET', key) return result } }) } // ── 演示：effect track/trigger ── const state = reactive({ a: 1, b: 2, c: 3 }) const e1Logs = [] effect(() => { e1Logs.push(state.a + state.b) }) console.log('effect1 依赖 a+b，初始值:', e1Logs.at(-1)) const e2Logs = [] effect(() => { e2Logs.push(state.b + state.c) }) console.log('effect2 依赖 b+c，初始值:', e2Logs.at(-1)) // 修改 a → 只触发 effect1 state.a = 10 console.log('修改 a=10 → effect1 触发:', e1Logs.at(-1), '| effect2 未触发:', e2Logs.length === 1) // 修改 b → 触发 effect1 和 effect2 state.b = 20 console.log('修改 b=20 → effect1:', e1Logs.at(-1), '| effect2:', e2Logs.at(-1)) // 修改 c → 只触发 effect2 state.c = 30 console.log('修改 c=30 → effect2 触发:', e2Logs.at(-1), '| effect1 未触发:', e1Logs.length === 3) console.log('\n依赖图：a→[effect1] | b→[effect1, effect2] | c→[effect2]')

3. ref / reactive / computed 实现

// 依赖收集基础（复用 track/trigger/effect） const targetMap = new WeakMap() let activeEffect = null const effectStack = [] /** @param {Object} target @param {string|symbol} key @returns {void} */ function track(target, key) { if (!activeEffect) return let depsMap = targetMap.get(target) if (!depsMap) targetMap.set(target, depsMap = new Map()) let dep = depsMap.get(key) if (!dep) depsMap.set(key, dep = new Set()) dep.add(activeEffect) } /** @param {Object} target @param {string} type @param {string|symbol} key @returns {void} */ function trigger(target, type, key) { const depsMap = targetMap.get(target) if (!depsMap) return const effects = new Set() depsMap.get(key)?.forEach(e => { if (e !== activeEffect) effects.add(e) }) effects.forEach(e => e.options?.scheduler ? e.options.scheduler(e) : e()) } /** @param {Function} fn @param {Object} [options={}] @returns {Function} */ function effect(fn, options = {}) { const effectFn = () => { effectStack.push(effectFn); activeEffect = effectFn const result = fn() effectStack.pop(); activeEffect = effectStack[effectStack.length - 1] ?? null return result } effectFn.deps = new Set(); effectFn.options = options if (!options.lazy) effectFn() return effectFn } /** * 包装基本类型为响应式引用，通过 .value 访问。 * @param {*} raw - 初始值 * @returns {{ __v_isRef: true, value: * }} ref 对象 */ function ref(raw) { const r = { __v_isRef: true, get value() { track(r, 'value'); return raw }, set value(newVal) { if (newVal !== raw) { raw = newVal; trigger(r, 'SET', 'value') } } } return r } /** * 创建带缓存的计算属性，依赖变化时标记为脏，读取时才重算。 * @param {Function} getter - 计算函数 * @returns {{ __v_isRef: true, value: * }} computed 对象 */ function computed(getter) { let dirty = true, _value const effectFn = effect(getter, { lazy: true, scheduler() { if (!dirty) { dirty = true; trigger(obj, 'SET', 'value') } } }) const obj = { __v_isRef: true, get value() { if (dirty) { _value = effectFn(); dirty = false } track(obj, 'value') return _value } } return obj } /** * 监听响应式数据变化，变化时执行回调。 * @param {Function|Object} source - 监听源（函数或 ref） * @param {Function} cb - 回调函数 (newVal, oldVal) * @param {Object} [options={}] - 配置项 * @param {boolean} [options.immediate] - 是否立即执行 * @returns {void} */ function watch(source, cb, options = {}) { let getter = typeof source === 'function' ? source : () => source.value let oldValue const job = () => { const newValue = effectFn() if (newValue !== oldValue) { cb(newValue, oldValue); oldValue = newValue } } const effectFn = effect(getter, { lazy: true, scheduler: job }) if (options.immediate) job() else oldValue = effectFn() } // ── 演示 ── // ref const count = ref(0) const logs = [] effect(() => { logs.push(`count=${count.value}`) }) count.value++; count.value++ console.log('ref count 变化:', logs.join(' → ')) // computed（带缓存） const price = ref(100) const quantity = ref(3) let computeCount = 0 const total = computed(() => { computeCount++; return price.value * quantity.value }) console.log('computed total:', total.value, '(计算次数:', computeCount, ')') console.log('再次读取 total（缓存）:', total.value, '(计算次数:', computeCount, ')') price.value = 200 console.log('price 变化后 total:', total.value, '(计算次数:', computeCount, ')') console.log('再次读取（缓存）:', total.value, '(计算次数:', computeCount, ')') // watch const name = ref('Alice') const watchLogs = [] watch(name, (newVal, oldVal) => { watchLogs.push(`${oldVal}→${newVal}`) }) name.value = 'Bob'; name.value = 'Charlie' console.log('watch 变化记录:', watchLogs.join(' | '))

4. Composition API vs Options API 面试高频

Vue2 Options API（按选项组织）
export default { data() { return { count: 0, name: '' } }, computed: { double() { return this.count * 2 } }, methods: { increment() { this.count++ } }, watch: { count(val) { console.log(val) } }, mounted() { /* 生命周期 */ } }
Vue3 Composition API（按逻辑组织）
import { ref, computed, watch, onMounted } from 'vue' export default { setup() { // 逻辑可抽取为 composable const count = ref(0) const double = computed(() => count.value * 2) const increment = () => count.value++ watch(count, val => console.log(val)) onMounted(() => { /* 生命周期 */ }) return { count, double, increment } } }
Composition API 优势： ① 相关逻辑聚合在一起（不像 Options API 分散在 data/methods/computed）； ② 逻辑可复用（抽取为 composable 函数，类似 React Hooks）； ③ 更好的 TypeScript 支持； ④ 树摇（Tree-shaking）友好
