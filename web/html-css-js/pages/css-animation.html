1. transition 过渡

/* 语法 */ transition: [property] [duration] [timing-function] [delay]; /* 示例 */ .btn { transition: all 0.3s ease; } .btn { transition: background-color 0.3s, transform 0.4s cubic-bezier(0.34,1.56,0.64,1); } /* timing-function 常用值 */ ease | ease-in | ease-out | ease-in-out | linear cubic-bezier(x1, y1, x2, y2) /* 自定义贝塞尔曲线 */

2. transform 变换

/* 2D 变换 */ transform: translate(50px, -20px); /* 位移 */ transform: scale(1.5); /* 缩放 */ transform: rotate(45deg); /* 旋转 */ transform: skew(15deg, 5deg); /* 倾斜 */ /* 3D 变换（父元素需设 perspective）*/ .parent { perspective: 400px; } .child { transform: rotateY(45deg); transform-style: preserve-3d; } /* 组合变换（从右到左执行）*/ transform: translateX(50px) rotate(30deg) scale(1.2);

性能提示 优先使用 transform 和 opacity 做动画，它们只触发合成（Composite），不触发重排（Reflow）或重绘（Repaint），性能最佳。

3. animation 关键帧动画

/* 定义关键帧 */ @keyframes bounce { 0%, 100% { transform: translateY(0); } 50% { transform: translateY(-40px); } } /* 使用动画 */ .ball { animation: bounce 1.2s ease-in-out infinite; /* name duration timing delay count direction fill-mode play-state */ } /* animation-fill-mode */ forwards /* 动画结束后保持最终状态 */ backwards /* 动画开始前应用第一帧 */ both /* 同时应用 forwards + backwards */

5. 渐变（Gradient）

/* 线性渐变 */ background: linear-gradient(135deg, #6c63ff, #00d4aa); background: linear-gradient(to right, red, yellow, green); /* 径向渐变 */ background: radial-gradient(circle at center, #6c63ff, transparent); /* 锥形渐变（饼图效果）*/ background: conic-gradient(#6c63ff 0deg 120deg, #00d4aa 120deg 240deg, #ff6584 240deg); /* 重复渐变（条纹）*/ background: repeating-linear-gradient(45deg, #6c63ff 0, #6c63ff 10px, transparent 10px, transparent 20px);
